## 1. 遷移概述

### 1.1 遷移目標

1. **代碼整合**：將3個重複的API文件整合為一個統一的模組化架構
2. **性能優化**：通過消除重複代碼和優化架構提升50%性能
3. **可維護性提升**：建立清晰的模組邊界和職責分離
4. **向後兼容**：確保現有API在遷移過程中正常工作

```python
# src/api/strategies/base.py
from abc import ABC, abstractmethod
from typing import TypeVar, Generic, Optional, List, Dict, Any
from pydantic import BaseModel

T = TypeVar('T', bound=BaseModel)

class AbstractBaseService(ABC, Generic[T]):
    """抽象基礎服務類，定義標準CRUD操作"""

    @abstractmethod
    async def create(self, data: dict) -> T:
        """創建實體"""
        pass

    @abstractmethod
    async def get(self, id: str) -> Optional[T]:
        """獲取實體"""
        pass

    @abstractmethod
    async def update(self, id: str, data: dict) -> T:
        """更新實體"""
        pass

    @abstractmethod
    async def delete(self, id: str) -> bool:
        """刪除實體"""
        pass

    @abstractmethod
    async def list(self, filters: Dict[str, Any] = None,
                   pagination: Dict[str, Any] = None) -> List[T]:
        """列表查詢"""
        pass

class ServiceRegistry:
    """服務註冊中心，管理所有服務實例"""

    def __init__(self):
        self._services: Dict[str, Any] = {}

    def register(self, name: str, service: Any):
        """註冊服務"""
        self._services[name] = service

    def get(self, name: str) -> Any:
        """獲取服務"""
        return self._services.get(name)

    def singleton(self, name: str):
        """單例裝飾器"""
        def decorator(cls):
            instance = None
            def wrapper(*args, **kwargs):
                nonlocal instance
                if instance is None:
                    instance = cls(*args, **kwargs)
                    self.register(name, instance)
                return instance
            return wrapper
        return decorator

# 全局服務註冊表
registry = ServiceRegistry()
```

#### 2.1 實現StrategyService (`services/strategy_service.py`)

```python
# src/api/strategies/services/strategy_service.py
from typing import List, Optional, Dict, Any
from ..base import AbstractBaseService, registry
from ..models import Strategy, StrategyPerformance
from ..repositories import IStrategyRepository
from .template_service import TemplateService
from .performance_service import PerformanceService
import logging

logger = logging.getLogger(__name__)

@registry.singleton("strategy_service")
class StrategyService(AbstractBaseService[Strategy]):
    """策略管理核心服務 - 整合三個文件的策略管理邏輯"""

    def __init__(
        self,
        repository: IStrategyRepository,
        template_service: TemplateService,
        performance_service: PerformanceService
    ):
        self.repository = repository
        self.template_service = template_service
        self.performance_service = performance_service

    async def create(self, data: dict) -> Strategy:
        """創建策略 - 統一三個文件的創建邏輯"""
        try:
            # 1. 生成策略ID
            strategy_id = await self._generate_strategy_id(data)

            # 2. 處理模板（如果有）
            parameters = await self._apply_template(data)

            # 3. 創建策略實例
            strategy = Strategy(
                id=strategy_id,
                name=data["name"],
                description=data["description"],
                strategy_type=data["strategy_type"],
                parameters=parameters,
                user_id=data.get("user_id")
            )

            # 4. 保存到數據庫
            saved_strategy = await self.repository.save(strategy)

            # 5. 初始化性能追蹤
            await self.performance_service.initialize_tracking(strategy_id)

            logger.info(f"策略創建成功: {strategy_id}")
            return saved_strategy

        except Exception as e:
            logger.error(f"創建策略失敗: {e}")
            raise

    async def get(self, id: str) -> Optional[Strategy]:
        """獲取策略"""
        return await self.repository.find_by_id(id)

    async def update(self, id: str, data: dict) -> Strategy:
        """更新策略"""
        # 檢查策略是否存在
        strategy = await self.get(id)
        if not strategy:
            raise ValueError(f"策略不存在: {id}")

        # 更新字段
        for key, value in data.items():
            if hasattr(strategy, key):
                setattr(strategy, key, value)

        strategy.updated_at = datetime.now()

        # 保存更新
        success = await self.repository.update(id, data)
        if not success:
            raise ValueError(f"更新策略失敗: {id}")

        return await self.get(id)

    async def delete(self, id: str) -> bool:
        """刪除策略"""
        # 檢查是否正在運行
        strategy = await self.get(id)
        if strategy and strategy.is_active:
            raise ValueError("無法刪除正在運行的策略")

        # 刪除相關數據
        await self.performance_service.cleanup_tracking(id)

        return await self.repository.delete(id)

    async def list(self, filters: Dict[str, Any] = None,
                   pagination: Dict[str, Any] = None) -> List[Strategy]:
        """列表查詢"""
        filters = filters or {}
        pagination = pagination or {"page": 1, "page_size": 20}

        return await self.repository.list_with_filters(filters, pagination)

    async def _generate_strategy_id(self, data: dict) -> str:
        """生成策略ID - 統一三個文件的ID生成邏輯"""
        from datetime import datetime
        user_id = data.get("user_id", "system")
        strategy_type = data.get("strategy_type", "unknown")
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        return f"{strategy_type}_{user_id}_{timestamp}"

    async def _apply_template(self, data: dict) -> Dict[str, Any]:
        """應用策略模板"""
        template_id = data.get("template_id")
        if template_id:
            template = await self.template_service.get_template(template_id)
            if template:
                # 合併模板參數和用戶參數
                template_params = template.get("parameters", {})
                user_params = data.get("parameters", {})
                return {**template_params, **user_params}

        return data.get("parameters", {})
```

```python
# src/api/strategies/services/execution_service.py
from typing import List, Optional, Dict, Any
from datetime import datetime
from ..base import registry
from ..models import Strategy, StrategySignal, StrategyPerformance
from .strategy_service import StrategyService
from ..repositories import ISignalRepository
import asyncio
import logging

logger = logging.getLogger(__name__)

@registry.singleton("execution_service")
class ExecutionService:
    """策略執行服務 - 整合執行邏輯"""

    def __init__(
        self,
        strategy_service: StrategyService,
        signal_repository: ISignalRepository
    ):
        self.strategy_service = strategy_service
        self.signal_repository = signal_repository
        self.running_executions: Dict[str, bool] = {}

    async def execute_strategy(
        self,
        strategy_id: str,
        execution_request: Dict[str, Any]
    ) -> Dict[str, Any]:
        """執行策略 - 整合三個文件的執行邏輯"""
        try:
            # 1. 獲取策略
            strategy = await self.strategy_service.get(strategy_id)
            if not strategy:
                raise ValueError(f"策略不存在: {strategy_id}")

            # 2. 生成執行ID
            execution_id = f"exec_{strategy_id}_{datetime.now().strftime('%Y%m%d_%H%M%S')}"

            # 3. 更新策略狀態
            await self.strategy_service.update(strategy_id, {
                "is_active": True,
                "last_executed": datetime.now()
            })

            # 4. 標記為正在運行
            self.running_executions[execution_id] = True

            # 5. 後台執行
            asyncio.create_task(
                self._execute_background(strategy, execution_id, execution_request)
            )

            return {
                "execution_id": execution_id,
                "strategy_id": strategy_id,
                "status": "running",
                "start_time": datetime.now()
            }

        except Exception as e:
            logger.error(f"執行策略失敗: {e}")
            raise

    async def stop_execution(self, execution_id: str) -> bool:
        """停止執行"""
        if execution_id in self.running_executions:
            self.running_executions[execution_id] = False
            return True
        return False

    async def get_execution_status(self, execution_id: str) -> Dict[str, Any]:
        """獲取執行狀態"""
        is_running = self.running_executions.get(execution_id, False)
        return {
            "execution_id": execution_id,
            "is_running": is_running
        }

    async def _execute_background(
        self,
        strategy: Strategy,
        execution_id: str,
        execution_request: Dict[str, Any]
    ):
        """後台執行策略"""
        try:
            logger.info(f"開始執行策略: {strategy.id}")

            # 模擬執行過程
            await asyncio.sleep(2)

            # 生成信號
            signals = await self._generate_signals(strategy, execution_id)

            # 保存信號
            for signal in signals:
                await self.signal_repository.save(signal)

            # 計算性能
            performance = await self._calculate_performance(signals)

            # 更新執行狀態
            self.running_executions[execution_id] = False

            logger.info(f"策略執行完成: {strategy.id}")

        except Exception as e:
            logger.error(f"策略執行失敗: {e}")
            self.running_executions[execution_id] = False

    async def _generate_signals(
        self,
        strategy: Strategy,
        execution_id: str
    ) -> List[StrategySignal]:
        """生成策略信號"""
        # 整合三個文件的信號生成邏輯
        signals = []
        base_time = datetime.now()

        # 根據策略類型生成不同的信號
        if strategy.strategy_type == "direct_rsi":
            signals = await self._generate_rsi_signals(strategy, execution_id, base_time)
        elif strategy.strategy_type == "sentiment_momentum":
            signals = await self._generate_sentiment_signals(strategy, execution_id, base_time)
        else:
            signals = await self._generate_technical_signals(strategy, execution_id, base_time)

        return signals

    async def _generate_rsi_signals(
        self,
        strategy: Strategy,
        execution_id: str,
        base_time: datetime
    ) -> List[StrategySignal]:
        """生成RSI策略信號"""
        # 從 personal_strategy_endpoints.py 移植並優化
        import random

        signals = []
        for i in range(10):
            signal = StrategySignal(
                signal_id=f"{execution_id}_rsi_{i+1:03d}",
                strategy_type=strategy.strategy_type,
                signal_type="buy" if i % 2 == 0 else "sell",
                strength=75.0 + random.uniform(-10, 10),
                confidence=80.0 + random.uniform(-10, 10),
                timestamp=base_time.replace(hour=i) if i < 24 else base_time,
                market_data={
                    "price": 150.0 + random.uniform(-10, 10),
                    "volume": 1000000 + random.uniform(-100000, 100000)
                },
                parameters=strategy.parameters,
                metadata={"execution_id": execution_id}
            )
            signals.append(signal)

        return signals
```

```python
from fastapi import APIRouter, HTTPException, Depends, Query
from typing import List, Optional, Dict, Any
from ..services import StrategyService
from ..models import Strategy
from ..schemas import StrategyCreateRequest, StrategyUpdateRequest

router = APIRouter(prefix="/api/strategies", tags=["策略管理"])

# 依賴注入
async def get_strategy_service() -> StrategyService:
    from ..base import registry
    return registry.get("strategy_service")

# CRUD路由 - 整合三個文件的基礎功能
@router.get("/", response_model=List[Strategy])
async def list_strategies(
    page: int = Query(1, ge=1),
    page_size: int = Query(20, ge=1, le=100),
    strategy_type: Optional[str] = Query(None),
    status: Optional[str] = Query(None),
    is_active: Optional[bool] = Query(None),
    service: StrategyService = Depends(get_strategy_service)
):
    """獲取策略列表 - 統一三個文件的列表邏輯"""
    filters = {}
    if strategy_type:
        filters["strategy_type"] = strategy_type
    if status:
        filters["status"] = status
    if is_active is not None:
        filters["is_active"] = is_active

    pagination = {"page": page, "page_size": page_size}

    return await service.list(filters, pagination)

@router.post("/", response_model=Strategy, status_code=201)
async def create_strategy(
    request: StrategyCreateRequest,
    service: StrategyService = Depends(get_strategy_service)
):
    """創建策略 - 統一三個文件的創建邏輯"""
    try:
        return await service.create(request.dict())
    except ValueError as e:
        raise HTTPException(status_code=400, detail=str(e))

@router.get("/{strategy_id}", response_model=Strategy)
async def get_strategy(
    strategy_id: str,
    service: StrategyService = Depends(get_strategy_service)
):
    """獲取策略詳情"""
    strategy = await service.get(strategy_id)
    if not strategy:
        raise HTTPException(status_code=404, detail="策略不存在")
    return strategy

@router.put("/{strategy_id}", response_model=Strategy)

    strategy_id: str,
    request: StrategyUpdateRequest,
    service: StrategyService = Depends(get_strategy_service)
):
    """更新策略"""
    try:
        return await service.update(strategy_id, request.dict(exclude_unset=True))
    except ValueError as e:
        raise HTTPException(status_code=404, detail=str(e))

@router.delete("/{strategy_id}", status_code=204)
async def delete_strategy(
    strategy_id: str,
    service: StrategyService = Depends(get_strategy_service)
):
    """刪除策略"""
    try:
        await service.delete(strategy_id)
    except ValueError as e:
        raise HTTPException(status_code=400, detail=str(e))
```

```python
def rollback_to_previous_version():
    """回滾到上一個版本"""
```
