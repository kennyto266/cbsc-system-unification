### Dashboard主类
```javascript
class Dashboard {
    constructor() {
        this.apiClient = new APIClient(API_BASE_URL);
        this.storage = new LocalStorage('strategy-dashboard');
        this.strategyManager = new StrategyManager(this.apiClient, this.storage);
        this.eventBus = new EventBus();
        this.isLoading = false;
        this.refreshInterval = null;
        this.currentTheme = 'light';

        this.init();
    }

    async init() {
        this.setupEventListeners();
        this.setupTheme();
        await this.loadData();
        this.startAutoRefresh();
        this.hideLoading();
    }

    setupEventListeners() {
        // 刷新按钮
        document.getElementById('refresh-btn').addEventListener('click', () => {
            this.refreshData();
        });

        // 主题切换
        document.getElementById('theme-toggle').addEventListener('click', () => {
            this.toggleTheme();
        });

        // 添加策略按钮
        document.getElementById('add-strategy-btn').addEventListener('click', () => {
            this.showStrategyForm();
        });

        // 时间范围选择
        document.getElementById('net-value-period').addEventListener('change', (e) => {
            this.updateCharts(e.target.value);
        });

        // 策略选择
        document.getElementById('drawdown-strategy').addEventListener('change', (e) => {
            this.updateDrawdownChart(e.target.value);
        });

        // 监听窗口大小变化
        window.addEventListener('resize', this.debounce(() => {
            this.handleResize();
        }, 250));
    }

    setupTheme() {
        const savedTheme = this.storage.get('theme') || 'light';
        this.setTheme(savedTheme);
    }

    async loadData() {
        this.showLoading();
        try {
            await this.strategyManager.loadStrategies();
            this.updateOverview();
            this.initializeComponents();
        } catch (error) {
            console.error('加载数据失败:', error);
            this.showError('加载数据失败，请检查网络连接');
        } finally {
            this.hideLoading();
        }
    }

    updateOverview() {
        const strategies = this.strategyManager.getStrategies();
        const runningStrategies = strategies.filter(s => s.status === 'running');

        // 计算总体指标
        const totalReturn = this.calculateTotalReturn(strategies);
        const avgSharpe = this.calculateAverageSharpe(strategies);
        const maxDrawdown = this.calculateMaxDrawdown(strategies);

        // 更新UI
        document.getElementById('total-return').textContent = (totalReturn * 100).toFixed(2) + '%';
        document.getElementById('avg-sharpe').textContent = avgSharpe.toFixed(2);
        document.getElementById('max-drawdown').textContent = (maxDrawdown * 100).toFixed(2) + '%';
        document.getElementById('running-strategies').textContent = `${runningStrategies.length}/4`;
        document.getElementById('running-count').textContent = runningStrategies.length;
        document.getElementById('stopped-count').textContent = strategies.length - runningStrategies.length;

        // 更新变化指示器
        this.updateChangeIndicators();
    }

    initializeComponents() {
        // 初始化策略列表
        const strategyListContainer = document.getElementById('strategy-list-container');
        this.strategyList = new StrategyList(strategyListContainer, this.strategyManager);

        // 初始化图表
        this.charts = {
            netValue: new NetValueChart('net-value-chart'),
            drawdown: new DrawdownChart('drawdown-chart'),
            heatmap: new MonthlyHeatmap('monthly-heatmap')
        };

        // 更新策略选择器
        this.updateStrategySelector();
    }

    updateStrategySelector() {
        const strategies = this.strategyManager.getStrategies();
        const select = document.getElementById('drawdown-strategy');

        // 保留"所有策略"选项
        select.innerHTML = '<option value="">所有策略</option>';

        // 添加策略选项
        strategies.forEach(strategy => {
            const option = document.createElement('option');
            option.value = strategy.id;
            option.textContent = strategy.name;
            select.appendChild(option);
        });
    }

    async refreshData() {
        this.showLoading();
        try {
            await this.loadData();
            this.showSuccess('数据刷新成功');
        } catch (error) {
            this.showError('刷新失败: ' + error.message);
        } finally {
            this.hideLoading();
        }
    }

    startAutoRefresh() {
        // 每30秒自动刷新
        this.refreshInterval = setInterval(() => {
            this.refreshData();
        }, 30000);
    }

    stopAutoRefresh() {
        if (this.refreshInterval) {
            clearInterval(this.refreshInterval);
            this.refreshInterval = null;
        }
    }

    showStrategyForm(mode = 'add', strategy = null) {
        const modal = document.getElementById('strategy-form-modal');
        const formContainer = modal.querySelector('.modal-content');

        this.strategyForm = new StrategyForm(formContainer, this.strategyManager);
        this.strategyForm.show(mode, strategy);
        modal.style.display = 'flex';
    }

    toggleTheme() {
        this.currentTheme = this.currentTheme === 'light' ? 'dark' : 'light';
        this.setTheme(this.currentTheme);
    }

    setTheme(theme) {
        document.body.className = theme;
        this.storage.set('theme', theme);

        // 更新主题图标
        const moonIcon = document.querySelector('.icon-moon');
        const sunIcon = document.querySelector('.icon-sun');

        if (theme === 'dark') {
            moonIcon.style.display = 'none';
            sunIcon.style.display = 'inline';
        } else {
            moonIcon.style.display = 'inline';
            sunIcon.style.display = 'none';
        }

        this.currentTheme = theme;
    }

    showLoading() {
        this.isLoading = true;
        document.getElementById('loading').style.display = 'flex';
    }

    hideLoading() {
        this.isLoading = false;
        document.getElementById('loading').style.display = 'none';
    }

    showSuccess(message) {
        this.showNotification(message, 'success');
    }

    showError(message) {
        this.showNotification(message, 'error');
    }

    showNotification(message, type) {
        const notifications = document.getElementById('notifications');
        const notification = document.createElement('div');
        notification.className = `notification notification-${type}`;
        notification.textContent = message;

        notifications.appendChild(notification);

        // 5秒后自动移除
        setTimeout(() => {
            notification.remove();
        }, 5000);
    }

    debounce(func, wait) {
        let timeout;
        return function executedFunction(...args) {
            const later = () => {
                clearTimeout(timeout);
                func(...args);
            };
            clearTimeout(timeout);
            timeout = setTimeout(later, wait);
        };
    }

    // 计算方法
    calculateTotalReturn(strategies) {
        if (strategies.length === 0) return 0;
        const totalValue = strategies.reduce((sum, s) => sum + (1 + s.current_return), 0);
        return totalValue / strategies.length - 1;
    }

    calculateAverageSharpe(strategies) {
        if (strategies.length === 0) return 0;
        const totalSharpe = strategies.reduce((sum, s) => sum + s.sharpe_ratio, 0);
        return totalSharpe / strategies.length;
    }

    calculateMaxDrawdown(strategies) {
        if (strategies.length === 0) return 0;
        return Math.max(...strategies.map(s => s.max_drawdown));
    }
}

// 应用启动
document.addEventListener('DOMContentLoaded', () => {
    const dashboard = new Dashboard();
});
```
