### 策略管理类
```javascript
class StrategyManager {
    constructor(apiClient, storage) {
        this.apiClient = apiClient;
        this.storage = storage;
        this.strategies = [];
        this.maxStrategies = 4;
        this.eventBus = new EventBus();
    }

    async loadStrategies() {
        // 从API加载策略列表
        try {
            this.strategies = await this.apiClient.getStrategies();
            this.cacheStrategies();
            this.eventBus.emit('strategies-loaded', this.strategies);
        } catch (error) {
            // 从缓存加载
            this.strategies = this.getCachedStrategies() || [];
            throw error;
        }
    }

    async addStrategy(strategyData) {
        if (this.strategies.length >= this.maxStrategies) {
            throw new Error(`最多支持${this.maxStrategies}个策略`);
        }

        try {
            const newStrategy = await this.apiClient.createStrategy(strategyData);
            this.strategies.push(newStrategy);
            this.cacheStrategies();
            this.eventBus.emit('strategy-added', newStrategy);
            return newStrategy;
        } catch (error) {
            this.eventBus.emit('error', error);
            throw error;
        }
    }

    async updateStrategy(id, updates) {
        try {
            const updatedStrategy = await this.apiClient.updateStrategy(id, updates);
            const index = this.strategies.findIndex(s => s.id === id);
            if (index !== -1) {
                this.strategies[index] = updatedStrategy;
                this.cacheStrategies();
                this.eventBus.emit('strategy-updated', updatedStrategy);
            }
            return updatedStrategy;
        } catch (error) {
            this.eventBus.emit('error', error);
            throw error;
        }
    }

    async deleteStrategy(id) {
        try {
            await this.apiClient.deleteStrategy(id);
            this.strategies = this.strategies.filter(s => s.id !== id);
            this.cacheStrategies();
            this.eventBus.emit('strategy-deleted', { id });
        } catch (error) {
            this.eventBus.emit('error', error);
            throw error;
        }
    }

    async startStrategy(id) {
        return await this.updateStrategyStatus(id, 'running');
    }

    async stopStrategy(id) {
        return await this.updateStrategyStatus(id, 'stopped');
    }

    getStrategies() {
        return this.strategies;
    }

    getStrategyById(id) {
        return this.strategies.find(s => s.id === id);
    }

    private cacheStrategies() {
        this.storage.set('strategies', this.strategies);
    }

    private getCachedStrategies() {
        return this.storage.get('strategies');
    }
}
```

### 策略列表组件
```javascript
class StrategyList {
    constructor(container, strategyManager) {
        this.container = container;
        this.strategyManager = strategyManager;
        this.render();
        this.bindEvents();
    }

    render() {
        const strategies = this.strategyManager.getStrategies();
        const html = `
            <div class="strategy-list">
                <div class="list-header">
                    <h2>策略列表</h2>
                    <button class="btn btn-primary" id="add-strategy-btn">
                        <i class="icon-plus"></i> 添加策略
                    </button>
                </div>
                <div class="strategy-cards" id="strategy-cards">
                    ${strategies.map(s => this.createStrategyCard(s)).join('')}
                </div>
            </div>
        `;
        this.container.innerHTML = html;
    }

    createStrategyCard(strategy) {
        const statusClass = strategy.status === 'running' ? 'status-running' : 'status-stopped';
        const returnClass = strategy.current_return >= 0 ? 'return-positive' : 'return-negative';

        return `
            <div class="strategy-card" data-id="${strategy.id}">
                <div class="card-header">
                    <h3>${strategy.name}</h3>
                    <span class="status ${statusClass}">${this.getStatusText(strategy.status)}</span>
                </div>
                <div class="card-metrics">
                    <div class="metric">
                        <span class="label">收益率</span>
                        <span class="value ${returnClass}">${(strategy.current_return * 100).toFixed(2)}%</span>
                    </div>
                    <div class="metric">
                        <span class="label">Sharpe比率</span>
                        <span class="value">${strategy.sharpe_ratio.toFixed(2)}</span>
                    </div>
                    <div class="metric">
                        <span class="label">最大回撤</span>
                        <span class="value">${(strategy.max_drawdown * 100).toFixed(2)}%</span>
                    </div>
                </div>
                <div class="card-actions">
                    <button class="btn btn-sm btn-primary" data-action="view" data-id="${strategy.id}">
                        查看详情
                    </button>
                    <button class="btn btn-sm ${strategy.status === 'running' ? 'btn-warning' : 'btn-success'}"
                            data-action="${strategy.status === 'running' ? 'stop' : 'start'}"
                            data-id="${strategy.id}">
                        ${strategy.status === 'running' ? '停止' : '启动'}
                    </button>
                    <button class="btn btn-sm btn-secondary" data-action="edit" data-id="${strategy.id}">
                        编辑
                    </button>
                    <button class="btn btn-sm btn-danger" data-action="delete" data-id="${strategy.id}">
                        删除
                    </button>
                </div>
            </div>
        `;
    }

    bindEvents() {
        this.container.addEventListener('click', (e) => {
            const button = e.target.closest('button');
            if (!button) return;

            const action = button.dataset.action;
            const id = button.dataset.id;

            switch (action) {
                case 'view':
                    this.viewStrategy(id);
                    break;
                case 'start':
                    this.startStrategy(id);
                    break;
                case 'stop':
                    this.stopStrategy(id);
                    break;
                case 'edit':
                    this.editStrategy(id);
                    break;
                case 'delete':
                    this.deleteStrategy(id);
                    break;
                case 'add':
                    this.addStrategy();
                    break;
            }
        });

        // 监听策略管理事件
        this.strategyManager.eventBus.on('strategies-loaded', () => {
            this.render();
        });

        this.strategyManager.eventBus.on('strategy-added', () => {
            this.render();
        });

        this.strategyManager.eventBus.on('strategy-updated', () => {
            this.render();
        });

        this.strategyManager.eventBus.on('strategy-deleted', () => {
            this.render();
        });
    }

    async viewStrategy(id) {
        const strategy = this.strategyManager.getStrategyById(id);
        if (strategy) {
            // 显示策略详情
            this.showStrategyDetail(strategy);
        }
    }

    async startStrategy(id) {
        try {
            await this.strategyManager.startStrategy(id);
            this.showSuccess('策略启动成功');
        } catch (error) {
            this.showError('启动策略失败: ' + error.message);
        }
    }

    async stopStrategy(id) {
        try {
            await this.strategyManager.stopStrategy(id);
            this.showSuccess('策略停止成功');
        } catch (error) {
            this.showError('停止策略失败: ' + error.message);
        }
    }

    editStrategy(id) {
        const strategy = this.strategyManager.getStrategyById(id);
        if (strategy) {
            // 显示编辑表单
            this.showEditForm(strategy);
        }
    }

    async deleteStrategy(id) {
        if (!confirm('确定要删除这个策略吗？')) return;

        try {
            await this.strategyManager.deleteStrategy(id);
            this.showSuccess('策略删除成功');
        } catch (error) {
            this.showError('删除策略失败: ' + error.message);
        }
    }

    addStrategy() {
        // 显示添加策略表单
        this.showAddForm();
    }

    showSuccess(message) {
        this.showNotification(message, 'success');
    }

    showError(message) {
        this.showNotification(message, 'error');
    }

    showNotification(message, type) {
        // 创建通知元素
        const notification = document.createElement('div');
        notification.className = `notification notification-${type}`;
        notification.textContent = message;

        document.body.appendChild(notification);

        // 3秒后自动移除
        setTimeout(() => {
            notification.remove();
        }, 3000);
    }
}
```

```javascript
class StrategyForm {
    constructor(container, strategyManager) {
        this.container = container;
        this.strategyManager = strategyManager;
        this.editingStrategy = null;
    }

    show(mode, strategy = null) {
        this.editingStrategy = mode === 'edit' ? strategy : null;
        this.render(mode, strategy);
        this.bindEvents();
    }

    render(mode, strategy) {
        const title = mode === 'edit' ? '编辑策略' : '添加新策略';
        const formData = strategy || this.getDefaultFormData();

        const html = `
            <div class="form-container">
                <div class="form-header">
                    <h2>${title}</h2>
                    <button class="btn-close" id="close-form">&times;</button>
                </div>
                <form id="strategy-form" class="strategy-form">
                    <div class="form-group">
                        <label for="strategy-name">策略名称</label>
                        <input type="text" id="strategy-name" name="name"
                               value="${formData.name}" required>
                    </div>

                    <div class="form-group">
                        <label for="strategy-description">策略描述</label>
                        <textarea id="strategy-description" name="description"
                                  rows="3">${formData.description}</textarea>
                    </div>

                    <div class="form-group">
                        <label for="initial-capital">初始资金</label>
                        <input type="number" id="initial-capital" name="initial_capital"
                               value="${formData.initial_capital}" min="10000" step="10000" required>
                    </div>

                    <div class="form-group">
                        <label for="risk-limit">风险限制（最大回撤）</label>
                        <input type="number" id="risk-limit" name="risk_limit"
                               value="${formData.risk_limit}" min="0" max="1" step="0.01" required>
                    </div>

                    <div class="form-group">
                        <label for="strategy-type">策略类型</label>
                        <select id="strategy-type" name="strategy_type" required>
                            <option value="momentum">动量策略</option>
                            <option value="mean_reversion">均值回归</option>
                            <option value="trend_following">趋势跟踪</option>
                            <option value="arbitrage">套利策略</option>
                        </select>
                    </div>

                    <div class="form-actions">
                        <button type="submit" class="btn btn-primary">
                            ${mode === 'edit' ? '更新' : '创建'}策略
                        </button>
                        <button type="button" class="btn btn-secondary" id="cancel-form">
                            取消
                        </button>
                    </div>
                </form>
            </div>
        `;

        this.container.innerHTML = html;
        this.container.style.display = 'block';
    }

    bindEvents() {
        const form = document.getElementById('strategy-form');
        const closeBtn = document.getElementById('close-form');
        const cancelBtn = document.getElementById('cancel-form');

        form.addEventListener('submit', async (e) => {
            e.preventDefault();
            await this.handleSubmit();
        });

        closeBtn.addEventListener('click', () => {
            this.hide();
        });

        cancelBtn.addEventListener('click', () => {
            this.hide();
        });
    }

    async handleSubmit() {
        const formData = new FormData(document.getElementById('strategy-form'));
        const data = Object.fromEntries(formData.entries());

        try {
            if (this.editingStrategy) {
                await this.strategyManager.updateStrategy(this.editingStrategy.id, data);
                this.showSuccess('策略更新成功');
            } else {
                await this.strategyManager.addStrategy(data);
                this.showSuccess('策略创建成功');
            }
            this.hide();
        } catch (error) {
            this.showError('保存失败: ' + error.message);
        }
    }

    hide() {
        this.container.style.display = 'none';
        this.editingStrategy = null;
    }

    showSuccess(message) {
        // 显示成功消息
    }

    showError(message) {
        // 显示错误消息
    }

    getDefaultFormData() {
        return {
            name: '',
            description: '',
            initial_capital: 100000,
            risk_limit: 0.2,
            strategy_type: 'momentum'
        };
    }
}
```
