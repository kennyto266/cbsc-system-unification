### 净值曲线图
```javascript
class NetValueChart {
    constructor(canvasId) {
        this.canvas = document.getElementById(canvasId);
        this.chart = null;
    }

    render(data, strategyName) {
        this.chart = new Chart(this.canvas, {
            type: 'line',
            data: {
                labels: data.timestamps,
                datasets: [{
                    label: strategyName,
                    data: data.cumulative_returns,
                    borderColor: 'rgb(75, 192, 192)',
                    backgroundColor: 'rgba(75, 192, 192, 0.1)',
                    tension: 0.1,
                    fill: true
                }]
            },
            options: {
                responsive: true,
                maintainAspectRatio: false,
                scales: {
                    y: {
                        beginAtZero: false,
                        ticks: {
                            callback: function(value) {
                                return (value * 100).toFixed(2) + '%';
                            }
                        }
                    }
                },
                plugins: {
                    tooltip: {
                        callbacks: {
                            label: function(context) {
                                return '净值: ' + (context.parsed.y * 100).toFixed(2) + '%';
                            }
                        }
                    }
                }
            }
        });
    }
}
```

### 月度收益热力图
```javascript
class MonthlyHeatmap {
    constructor(containerId) {
        this.container = document.getElementById(containerId);
    }

    render(data) {
        // 使用Canvas或SVG实现热力图
        const heatmapData = this.processMonthlyData(data);
        this.createHeatmapGrid(heatmapData);
    }

    processMonthlyData(data) {
        // 将日度数据按月聚合
        const monthlyData = {};

        data.timestamps.forEach((timestamp, index) => {
            const date = new Date(timestamp);
            const monthKey = `${date.getFullYear()}-${String(date.getMonth() + 1).padStart(2, '0')}`;

            if (!monthlyData[monthKey]) {
                monthlyData[monthKey] = [];
            }

            monthlyData[monthKey].push(data.daily_returns[index]);
        });

        return monthlyData;
    }

    createHeatmapGrid(data) {
        // 创建热力图网格
        const grid = document.createElement('div');
        grid.className = 'heatmap-grid';

        Object.entries(data).forEach(([month, returns]) => {
            const monthlyReturn = returns.reduce((sum, r) => sum + r, 0);
            const cell = document.createElement('div');
            cell.className = 'heatmap-cell';
            cell.textContent = (monthlyReturn * 100).toFixed(2) + '%';
            cell.style.backgroundColor = this.getColorForValue(monthlyReturn);
            grid.appendChild(cell);
        });

        this.container.innerHTML = '';
        this.container.appendChild(grid);
    }

    getColorForValue(value) {
        // 根据收益值返回颜色
        if (value > 0.05) return '#22c55e'; // 绿色
        if (value > 0) return '#84cc16'; // 浅绿
        if (value > -0.02) return '#eab308'; // 黄色
        return '#ef4444'; // 红色
    }
}
```

### 响应式图表配置
```javascript
const responsiveConfig = {
    mobile: {
        width: '100%',
        height: 300,
        fontSize: 12
    },
    tablet: {
        width: '100%',
        height: 400,
        fontSize: 14
    },
    desktop: {
        width: '100%',
        height: 500,
        fontSize: 16
    }
};

function getResponsiveConfig() {
    const width = window.innerWidth;
    if (width < 768) return responsiveConfig.mobile;
    if (width < 1024) return responsiveConfig.tablet;
    return responsiveConfig.desktop;
}
```
