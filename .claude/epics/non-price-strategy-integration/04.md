class RateLimiter:
    """Advanced rate limiting with Redis backend"""

    def __init__(self):
        self.redis_client = redis.Redis(
            host='localhost',
            port=6379,
            db=1,
            decode_responses=True
        )

        # Rate limiting configurations
        self.limits = {
            'api_calls_per_minute': 60,  # Default limit
            'real_time_connections': 5,
            'data_exports_per_hour': 10,
            'strategy_executions_per_hour': 100
        }

        self.redis_key_prefix = "rate_limit"

    async def check_rate_limit(
        self,
        user_id: str,
        limit_type: str,
        custom_limit: Optional[int] = None
    ) -> bool:
        """Check if user has exceeded rate limit"""

        limit = custom_limit or self.limits.get(limit_type, self.limits['api_calls_per_minute'])
        key = f"{self.redis_key_prefix}:{limit_type}:{user_id}"

        try:
            # Get current count
            current_count = self.redis_client.get(key) or 0
            current_count = int(current_count)

            # Check if limit exceeded
            if current_count >= limit:
                # Log rate limit violation
                await self._log_rate_limit_violation(user_id, limit_type, limit, current_count)
                return False

            # Increment counter with expiration
            pipe = self.redis_client.pipeline()
            pipe.incr(key)
            pipe.expire(key, 60)  # 1 minute expiration
            pipe.execute()

            return True

        except Exception as e:
            logging.error(f"Rate limiter error: {e}")
            # Fail open - allow request if rate limiter fails
            return True

    async def get_remaining_requests(
        self,
        user_id: str,
        limit_type: str
    ) -> int:
        """Get remaining requests for user"""

        limit = self.limits.get(limit_type, self.limits['api_calls_per_minute'])
        key = f"{self.redis_key_prefix}:{limit_type}:{user_id}"

        try:
            current_count = int(self.redis_client.get(key) or 0)
            remaining = max(0, limit - current_count)
            return remaining

        except Exception as e:
            logging.error(f"Failed to get remaining requests: {e}")
            return limit  # Assume full limit if error

class DataEncryption:
    """Encryption service for sensitive financial data"""

    def __init__(self):
        self.key = self._load_or_generate_key()
        self.fernet = Fernet(self.key)
        self.logger = logging.getLogger(__name__)

    def _load_or_generate_key(self) -> bytes:
        """Load encryption key from environment or generate new one"""

        # Try to load from environment
        key_b64 = os.getenv('CBSC_ENCRYPTION_KEY')
        if key_b64:
            try:
                return base64.urlsafe_b64decode(key_b64.encode())
            except Exception as e:
                self.logger.error(f"Failed to decode encryption key: {e}")

        # Generate new key
        self.logger.warning("Generating new encryption key - please set CBSC_ENCRYPTION_KEY environment variable")
        key = Fernet.generate_key()

        # Store in environment for persistence (in production, use secure key management)
        os.environ['CBSC_ENCRYPTION_KEY'] = base64.urlsafe_b64encode(key).decode()

        return key

    def encrypt_data(self, data: Any) -> str:
        """Encrypt sensitive data"""

        try:
            # Convert data to JSON if it's a dictionary or object
            if isinstance(data, (dict, list)):
                data_str = json.dumps(data, default=str)
            else:
                data_str = str(data)

            encrypted_data = self.fernet.encrypt(data_str.encode())
            return base64.urlsafe_b64encode(encrypted_data).decode()

        except Exception as e:
            self.logger.error(f"Failed to encrypt data: {e}")
            raise ValueError("Encryption failed")

    def decrypt_data(self, encrypted_data: str) -> Any:
        """Decrypt sensitive data"""

        try:
            encrypted_bytes = base64.urlsafe_b64decode(encrypted_data.encode())
            decrypted_data = self.fernet.decrypt(encrypted_bytes).decode()

            # Try to parse as JSON first
            try:
                return json.loads(decrypted_data)
            except json.JSONDecodeError:
                # Return as string if not valid JSON
                return decrypted_data

        except Exception as e:
            self.logger.error(f"Failed to decrypt data: {e}")
            raise ValueError("Decryption failed")

    def encrypt_sensitive_macro_data(self, macro_data: Dict[str, Any]) -> Dict[str, Any]:
        """Encrypt sensitive macroeconomic indicators while keeping public fields visible"""

        # Define which fields are sensitive
        sensitive_fields = {
            'exact_values',  # Exact monetary values
            'internal_notes',  # Internal analysis notes
            'source_details'   # Detailed source information
        }

        encrypted_data = macro_data.copy()
        data_to_encrypt = {}

        # Extract sensitive data
        for field in sensitive_fields:
            if field in macro_data:
                data_to_encrypt[field] = macro_data[field]
                del encrypted_data[field]

        # Encrypt sensitive data and store as encrypted field
        if data_to_encrypt:
            encrypted_data['sensitive_data_encrypted'] = self.encrypt_data(data_to_encrypt)

        return encrypted_data

    def decrypt_sensitive_macro_data(self, encrypted_macro_data: Dict[str, Any]) -> Dict[str, Any]:
        """Decrypt sensitive macroeconomic indicators"""

        decrypted_data = encrypted_macro_data.copy()

        if 'sensitive_data_encrypted' in encrypted_macro_data:
            try:
                sensitive_data = self.decrypt_data(encrypted_data['sensitive_data_encrypted'])
                decrypted_data.update(sensitive_data)
                del decrypted_data['sensitive_data_encrypted']
            except Exception as e:
                self.logger.error(f"Failed to decrypt macro data: {e}")

        return decrypted_data

    def hash_sensitive_field(self, field_value: str) -> str:
        """Create irreversible hash for sensitive field comparisons"""

        import hashlib
        import hmac

        # Create HMAC with pepper
        pepper = os.getenv('HASH_PEPPER', 'default_pepper_change_in_production')
        h = hmac.new(
            pepper.encode(),
            field_value.encode(),
            hashlib.sha256
        )
        return h.hexdigest()

    async def generate_compliance_report(
        self,
        start_date: datetime,
        end_date: datetime,
        report_type: str = "security"
    ) -> Dict[str, Any]:
        """Generate compliance report for audit period"""

        try:
            # Fetch relevant audit logs
            audit_logs = await self.search_audit_logs(start_date, end_date)

            # Generate report based on type
            if report_type == "security":
                return self._generate_security_report(audit_logs, start_date, end_date)
            elif report_type == "access":
                return self._generate_access_report(audit_logs, start_date, end_date)
            elif report_type == "data":
                return self._generate_data_access_report(audit_logs, start_date, end_date)
            else:
                return self._generate_general_report(audit_logs, start_date, end_date)

        except Exception as e:
            self.logger.error(f"Failed to generate compliance report: {e}")
            return {"error": str(e)}

    def _generate_security_report(
        self,
        audit_logs: list,
        start_date: datetime,
        end_date: datetime
    ) -> Dict[str, Any]:
        """Generate security-focused compliance report"""

        security_events = [
            log for log in audit_logs
            if log.get('severity') in ['critical', 'high']
        ]

        security_violations = [
            log for log in security_events
            if log.get('event_type') == AuditEventType.SECURITY_VIOLATION.value
        ]

        failed_logins = [
            log for log in audit_logs
            if log.get('event_type') == AuditEventType.USER_LOGIN.value
            and log.get('details', {}).get('success') is False
        ]

        permission_denials = [
            log for log in audit_logs
            if log.get('event_type') == AuditEventType.PERMISSION_DENIED.value
        ]

        return {
            "report_type": "security",
            "period": {
                "start_date": start_date.isoformat(),
                "end_date": end_date.isoformat()
            },
            "summary": {
                "total_events": len(audit_logs),
                "security_events": len(security_events),
                "security_violations": len(security_violations),
                "failed_logins": len(failed_logins),
                "permission_denials": len(permission_denials)
            },
            "details": {
                "security_violations_by_type": self._group_by_field(security_violations, "details.violation_type"),
                "failed_login_ips": list(set(log.get("ip_address") for log in failed_logins)),
                "permission_denials_by_resource": self._group_by_field(permission_denials, "details.resource"),
                "high_risk_activities": [log for log in security_events if log.get("severity") == "critical"]
            },
            "recommendations": self._generate_security_recommendations(security_violations)
        }

    def _group_by_field(self, items: list, field_path: str) -> Dict[str, int]:
        """Group items by a nested field"""

        groups = {}
        for item in items:
            try:
                # Navigate nested field path (e.g., "details.violation_type")
                value = item
                for field in field_path.split('.'):
                    if isinstance(value, dict) and field in value:
                        value = value[field]
                    else:
                        value = None
                        break

                if value:
                    groups[str(value)] = groups.get(str(value), 0) + 1

            except Exception:
                continue

        return groups

    def _generate_security_recommendations(self, violations: list) -> list:
        """Generate security recommendations based on violations"""

        recommendations = []

        violation_types = set()
        for violation in violations:
            if 'details' in violation:
                violation_type = violation['details'].get('violation_type')
                if violation_type:
                    violation_types.add(violation_type)

        # Generate recommendations based on violation patterns
        if 'unauthorized_access' in violation_types:
            recommendations.append("Review and strengthen access control policies")

        if 'data_exfiltration' in violation_types:
            recommendations.append("Implement data loss prevention (DLP) measures")

        if 'suspicious_activity' in violation_types:
            recommendations.append("Implement real-time behavioral analysis")

        if 'rate_limit_exceeded' in violation_types:
            recommendations.append("Review and adjust rate limiting thresholds")

        if not recommendations:
            recommendations.append("Continue monitoring security events")

        return recommendations

    async def start_background_flusher(self) -> None:
        """Start background task to flush audit buffer periodically"""

        while True:
            try:
                await asyncio.sleep(self.flush_interval)
                await self._flush_buffer()

            except Exception as e:
                self.logger.error(f"Background audit flusher error: {e}")
                await asyncio.sleep(60)  # Wait before retrying
