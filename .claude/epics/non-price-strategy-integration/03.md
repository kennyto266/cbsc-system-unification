```python
from abc import ABC, abstractmethod
from typing import Dict, List, Any, Optional
from datetime import datetime
import asyncio
import pandas as pd
from dataclasses import dataclass

@dataclass
class UnifiedDataPoint:
    timestamp: datetime
    symbol: str
    source: str  # 'price', 'hkma', 'sentiment', 'alternative'
    data_type: str
    value: float
    metadata: Dict[str, Any]
    quality_score: float
    is_valid: bool

class DataSource(ABC):
    """Abstract base class for all data sources"""

    @abstractmethod
    async def fetch_data(self, symbol: str, start_time: datetime, end_time: datetime) -> List[UnifiedDataPoint]:
        pass

    @abstractmethod
    async def validate_data_quality(self, data: List[UnifiedDataPoint]) -> Dict[str, Any]:
        pass

class PriceDataSource(DataSource):
    """Yahoo Finance price data source"""

    async def fetch_data(self, symbol: str, start_time: datetime, end_time: datetime) -> List[UnifiedDataPoint]:
        # Implementation for price data fetching
        pass

class HKMADataSource(DataSource):
    """HKMA macroeconomic data source"""

    async def fetch_data(self, symbol: str, start_time: datetime, end_time: datetime) -> List[UnifiedDataPoint]:
        # Implementation for HKMA data fetching
        pass

class SentimentDataSource(DataSource):
    """Sentiment analysis data source"""

    async def fetch_data(self, symbol: str, start_time: datetime, end_time: datetime) -> List[UnifiedDataPoint]:
        # Implementation for sentiment data fetching
        pass

class UnifiedDataPipeline:
    """Coordinates data ingestion from all sources"""

    def __init__(self):
        self.data_sources: Dict[str, DataSource] = {
            'price': PriceDataSource(),
            'hkma': HKMADataSource(),
            'sentiment': SentimentDataSource()
        }
        self.quality_validator = DataQualityValidator()
        self.cache_manager = UnifiedCacheManager()

    async def fetch_unified_data(
        self,
        symbols: List[str],
        start_time: datetime,
        end_time: datetime,
        sources: Optional[List[str]] = None
    ) -> Dict[str, List[UnifiedDataPoint]]:
        """Fetch synchronized data from multiple sources"""

        if sources is None:
            sources = list(self.data_sources.keys())

        tasks = []
        for symbol in symbols:
            for source_name in sources:
                if source_name in self.data_sources:
                    task = self._fetch_and_validate_source_data(
                        symbol, source_name, start_time, end_time
                    )
                    tasks.append(task)

        results = await asyncio.gather(*tasks, return_exceptions=True)

        # Process results and merge data
        unified_data = self._merge_data_sources(results, symbols)

        # Cache unified data
        await self.cache_manager.cache_unified_data(unified_data)

        return unified_data

    async def _fetch_and_validate_source_data(
        self,
        symbol: str,
        source_name: str,
        start_time: datetime,
        end_time: datetime
    ) -> Tuple[str, str, List[UnifiedDataPoint]]:
        """Fetch and validate data from a single source"""

        source = self.data_sources[source_name]
        raw_data = await source.fetch_data(symbol, start_time, end_time)
        validation_result = await source.validate_data_quality(raw_data)

        # Mark invalid data points
        for data_point in raw_data:
            if validation_result.get('overall_quality', 1.0) < 0.8:
                data_point.is_valid = False

        return (symbol, source_name, raw_data)

    def _merge_data_sources(
        self,
        results: List[Any],
        symbols: List[str]
    ) -> Dict[str, List[UnifiedDataPoint]]:
        """Merge data from different sources with timestamp alignment"""

        unified_data = {}

        for symbol in symbols:
            symbol_data = []

            for result in results:
                if isinstance(result, Exception):
                    continue

                result_symbol, source_name, data_points = result
                if result_symbol == symbol:
                    symbol_data.extend(data_points)

            # Sort by timestamp and remove duplicates
            symbol_data.sort(key=lambda x: x.timestamp)
            unified_data[symbol] = symbol_data

        return unified_data
```

```python
import redis
import pickle
import json
from typing import Any, Optional, Dict, List
from datetime import datetime, timedelta
import asyncio
import logging

class UnifiedCacheManager:
    """Manages L1 (memory) and L2 (Redis) caching for all data types"""

    def __init__(self):
        self.l1_cache: Dict[str, Any] = {}  # Memory cache
        self.l1_ttl: Dict[str, datetime] = {}  # Memory cache TTL
        self.l1_max_size = 1000  # Maximum items in memory cache

        # Redis connection (L2 cache)
        self.redis_client = redis.Redis(
            host='localhost',
            port=6379,
            db=0,
            decode_responses=False
        )

        # Cache configuration
        self.cache_ttl = {
            'price_data': timedelta(minutes=5),
            'hkma_data': timedelta(hours=1),
            'sentiment_data': timedelta(minutes=15),
            'unified_data': timedelta(minutes=10),
            'backtest_results': timedelta(days=1)
        }

        self.logger = logging.getLogger(__name__)

    async def get(self, cache_key: str, cache_type: str = 'unified_data') -> Optional[Any]:
        """Get data from cache (L1 first, then L2)"""

        # Check L1 cache first
        if cache_key in self.l1_cache:
            if self._is_l1_valid(cache_key):
                self.logger.debug(f"L1 cache hit: {cache_key}")
                return self.l1_cache[cache_key]
            else:
                # Remove expired L1 entry
                del self.l1_cache[cache_key]
                del self.l1_ttl[cache_key]

        # Check L2 cache (Redis)
        try:
            cached_data = await self._get_from_redis(cache_key, cache_type)
            if cached_data is not None:
                self.logger.debug(f"L2 cache hit: {cache_key}")
                # Promote to L1 cache
                await self._store_in_l1(cache_key, cached_data, cache_type)
                return cached_data
        except Exception as e:
            self.logger.error(f"Redis cache error: {e}")

        self.logger.debug(f"Cache miss: {cache_key}")
        return None

    async def set(
        self,
        cache_key: str,
        data: Any,
        cache_type: str = 'unified_data',
        custom_ttl: Optional[timedelta] = None
    ) -> None:
        """Store data in both L1 and L2 cache"""

        # Store in L1 cache
        await self._store_in_l1(cache_key, data, cache_type, custom_ttl)

        # Store in L2 cache
        await self._store_in_redis(cache_key, data, cache_type, custom_ttl)

    async def _store_in_l1(
        self,
        cache_key: str,
        data: Any,
        cache_type: str,
        custom_ttl: Optional[timedelta] = None
    ) -> None:
        """Store data in L1 (memory) cache"""

        # Implement LRU eviction if cache is full
        if len(self.l1_cache) >= self.l1_max_size:
            await self._evict_l1_oldest()

        ttl = custom_ttl or self.cache_ttl.get(cache_type, timedelta(minutes=10))

        self.l1_cache[cache_key] = data
        self.l1_ttl[cache_key] = datetime.now() + ttl

    def _is_l1_valid(self, cache_key: str) -> bool:
        """Check if L1 cache entry is still valid"""
        if cache_key not in self.l1_ttl:
            return False

        return datetime.now() < self.l1_ttl[cache_key]

    async def _evict_l1_oldest(self) -> None:
        """Evict oldest entries from L1 cache"""
        oldest_key = min(self.l1_ttl.keys(), key=lambda k: self.l1_ttl[k])
        del self.l1_cache[oldest_key]
        del self.l1_ttl[oldest_key]

    async def _get_from_redis(self, cache_key: str, cache_type: str) -> Optional[Any]:
        """Get data from Redis (L2 cache)"""
        try:
            redis_key = f"{cache_type}:{cache_key}"
            cached_data = self.redis_client.get(redis_key)

            if cached_data:
                # Deserialize based on cache type
                if cache_type == 'backtest_results':
                    return pickle.loads(cached_data)
                else:
                    return json.loads(cached_data)

            return None

        except Exception as e:
            self.logger.error(f"Redis get error: {e}")
            return None

    async def _store_in_redis(
        self,
        cache_key: str,
        data: Any,
        cache_type: str,
        custom_ttl: Optional[timedelta] = None
    ) -> None:
        """Store data in Redis (L2 cache)"""
        try:
            redis_key = f"{cache_type}:{cache_key}"

            # Serialize based on cache type
            if cache_type == 'backtest_results':
                serialized_data = pickle.dumps(data)
            else:
                serialized_data = json.dumps(data, default=str)

            ttl_seconds = int(
                (custom_ttl or self.cache_ttl.get(cache_type, timedelta(minutes=10))).total_seconds()
            )

            self.redis_client.setex(redis_key, ttl_seconds, serialized_data)

        except Exception as e:
            self.logger.error(f"Redis set error: {e}")

    async def invalidate_pattern(self, pattern: str) -> None:
        """Invalidate cache entries matching a pattern"""
        # Invalidate L1 cache
        keys_to_remove = [key for key in self.l1_cache.keys() if pattern in key]
        for key in keys_to_remove:
            del self.l1_cache[key]
            if key in self.l1_ttl:
                del self.l1_ttl[key]

        # Invalidate L2 cache
        try:
            for key in self.redis_client.scan_iter(match=f"*{pattern}*"):
                self.redis_client.delete(key)
        except Exception as e:
            self.logger.error(f"Redis pattern invalidation error: {e}")
```

**3. Data Quality Validation (`src/data/quality_validator.py`)**
```python
from typing import Dict, List, Any, Tuple
from datetime import datetime
import numpy as np
import pandas as pd
from enum import Enum

class QualityLevel(Enum):
    EXCELLENT = 5
    GOOD = 4
    ACCEPTABLE = 3
    POOR = 2
    UNACCEPTABLE = 1

class DataQualityIssue(Enum):
    MISSING_VALUES = "missing_values"
    OUTLIERS = "outliers"
    STALE_DATA = "stale_data"
    INCONSISTENT_TIMESTAMPS = "inconsistent_timestamps"
    DUPLICATE_DATA = "duplicate_data"
    INVALID_FORMAT = "invalid_format"
    SOURCE_UNAVAILABLE = "source_unavailable"

class DataQualityValidator:
    """Validates data quality across all sources"""

    def __init__(self):
        self.quality_thresholds = {
            'completeness_threshold': 0.95,  # 95% completeness required
            'outlier_threshold': 3.0,        # 3-sigma outlier detection
            'staleness_threshold': 300,       # 5 minutes staleness threshold
            'duplicate_threshold': 0.01       # 1% duplicate threshold
        }

    async def validate_unified_data(
        self,
        data: List[UnifiedDataPoint],
        source_type: str
    ) -> Dict[str, Any]:
        """Comprehensive data quality validation"""

        if not data:
            return {
                'overall_quality': QualityLevel.UNACCEPTABLE.value,
                'quality_score': 0.0,
                'issues': [DataQualityIssue.MISSING_VALUES.value],
                'recommendations': ['No data available for validation']
            }

        validation_results = {
            'source_type': source_type,
            'total_points': len(data),
            'valid_points': len([dp for dp in data if dp.is_valid]),
            'quality_checks': {},
            'issues': [],
            'recommendations': []
        }

        # Run quality checks
        validation_results['quality_checks']['completeness'] = self._check_completeness(data)
        validation_results['quality_checks']['outliers'] = self._check_outliers(data)
        validation_results['quality_checks']['staleness'] = self._check_staleness(data)
        validation_results['quality_checks']['consistency'] = self._check_timestamp_consistency(data)
        validation_results['quality_checks']['duplicates'] = self._check_duplicates(data)

        # Calculate overall quality score
        quality_score = self._calculate_quality_score(validation_results['quality_checks'])
        validation_results['quality_score'] = quality_score
        validation_results['overall_quality'] = self._quality_score_to_level(quality_score).value

        # Generate recommendations
        validation_results['recommendations'] = self._generate_recommendations(
            validation_results['quality_checks'],
            source_type
        )

        # Log quality issues
        if quality_score < 0.8:
            self._log_quality_issues(validation_results)

        return validation_results

    def _check_completeness(self, data: List[UnifiedDataPoint]) -> Dict[str, Any]:
        """Check data completeness"""

        total_points = len(data)
        valid_points = len([dp for dp in data if dp.is_valid])
        completeness_rate = valid_points / total_points if total_points > 0 else 0

        issues = []
        if completeness_rate < self.quality_thresholds['completeness_threshold']:
            issues.append(f"Low completeness: {completeness_rate:.2%}")

        return {
            'completeness_rate': completeness_rate,
            'total_points': total_points,
            'valid_points': valid_points,
            'issues': issues,
            'passes_threshold': completeness_rate >= self.quality_thresholds['completeness_threshold']
        }

    def _check_outliers(self, data: List[UnifiedDataPoint]) -> Dict[str, Any]:
        """Check for statistical outliers"""

        if len(data) < 10:
            return {
                'outliers_detected': 0,
                'outlier_percentage': 0.0,
                'issues': ['Insufficient data for outlier detection'],
                'passes_threshold': False
            }

        values = [dp.value for dp in data if dp.is_valid]
        if not values:
            return {
                'outliers_detected': 0,
                'outlier_percentage': 0.0,
                'issues': ['No valid data points for outlier detection'],
                'passes_threshold': False
            }

        mean_val = np.mean(values)
        std_val = np.std(values)

        outliers = []
        for i, value in enumerate(values):
            z_score = abs((value - mean_val) / std_val) if std_val > 0 else 0
            if z_score > self.quality_thresholds['outlier_threshold']:
                outliers.append(i)

        outlier_percentage = len(outliers) / len(values)

        issues = []
        if outlier_percentage > 0.05:  # More than 5% outliers
            issues.append(f"High outlier rate: {outlier_percentage:.2%}")

        return {
            'outliers_detected': len(outliers),
            'outlier_percentage': outlier_percentage,
            'outlier_indices': outliers,
            'mean': mean_val,
            'std': std_val,
            'issues': issues,
            'passes_threshold': outlier_percentage <= 0.05
        }

    def _check_staleness(self, data: List[UnifiedDataPoint]) -> Dict[str, Any]:
        """Check data freshness"""

        if not data:
            return {
                'age_seconds': float('inf'),
                'is_stale': True,
                'issues': ['No data available'],
                'passes_threshold': False
            }

        latest_timestamp = max(dp.timestamp for dp in data)
        current_time = datetime.now()
        age_seconds = (current_time - latest_timestamp).total_seconds()

        is_stale = age_seconds > self.quality_thresholds['staleness_threshold']

        issues = []
        if is_stale:
            issues.append(f"Data is {age_seconds:.0f} seconds old")

        return {
            'age_seconds': age_seconds,
            'latest_timestamp': latest_timestamp,
            'is_stale': is_stale,
            'issues': issues,
            'passes_threshold': not is_stale
        }

    def _check_timestamp_consistency(self, data: List[UnifiedDataPoint]) -> Dict[str, Any]:
        """Check timestamp consistency and ordering"""

        if len(data) < 2:
            return {
                'is_consistent': True,
                'issues': ['Insufficient data for consistency check'],
                'passes_threshold': True
            }

        timestamps = [dp.timestamp for dp in data]
        timestamps_sorted = sorted(timestamps)

        # Check if timestamps are properly ordered
        is_ordered = timestamps == timestamps_sorted

        # Check for duplicate timestamps
        duplicate_timestamps = len(timestamps) - len(set(timestamps))
        duplicate_percentage = duplicate_timestamps / len(timestamps)

        issues = []
        if not is_ordered:
            issues.append("Timestamps are not properly ordered")

        if duplicate_percentage > self.quality_thresholds['duplicate_threshold']:
            issues.append(f"High duplicate timestamp rate: {duplicate_percentage:.2%}")

        return {
            'is_consistent': is_ordered and duplicate_percentage <= self.quality_thresholds['duplicate_threshold'],
            'ordered_correctly': is_ordered,
            'duplicate_timestamps': duplicate_timestamps,
            'duplicate_percentage': duplicate_percentage,
            'issues': issues,
            'passes_threshold': is_ordered and duplicate_percentage <= self.quality_thresholds['duplicate_threshold']
        }

    def _check_duplicates(self, data: List[UnifiedDataPoint]) -> Dict[str, Any]:
        """Check for duplicate data points"""

        if len(data) < 2:
            return {
                'duplicates_found': 0,
                'duplicate_percentage': 0.0,
                'issues': ['Insufficient data for duplicate check'],
                'passes_threshold': True
            }

        # Create hashable representation of data points
        data_hashes = []
        for dp in data:
            hash_repr = (
                dp.timestamp.isoformat(),
                dp.symbol,
                dp.source,
                dp.data_type,
                str(dp.value)
            )
            data_hashes.append(hash(hash_repr))

        unique_hashes = set(data_hashes)
        duplicates = len(data_hashes) - len(unique_hashes)
        duplicate_percentage = duplicates / len(data_hashes)

        issues = []
        if duplicate_percentage > self.quality_thresholds['duplicate_threshold']:
            issues.append(f"High duplicate rate: {duplicate_percentage:.2%}")

        return {
            'duplicates_found': duplicates,
            'duplicate_percentage': duplicate_percentage,
            'total_points': len(data),
            'unique_points': len(unique_hashes),
            'issues': issues,
            'passes_threshold': duplicate_percentage <= self.quality_thresholds['duplicate_threshold']
        }

    def _calculate_quality_score(self, quality_checks: Dict[str, Any]) -> float:
        """Calculate overall quality score from individual checks"""

        weights = {
            'completeness': 0.3,
            'outliers': 0.2,
            'staleness': 0.2,
            'consistency': 0.2,
            'duplicates': 0.1
        }

        scores = {}

        # Completeness score
        completeness = quality_checks.get('completeness', {})
        scores['completeness'] = completeness.get('completeness_rate', 0.0)

        # Outliers score (inverse relationship with outlier percentage)
        outliers = quality_checks.get('outliers', {})
        outlier_penalty = outliers.get('outlier_percentage', 0.0) * 2  # Double penalty
        scores['outliers'] = max(0.0, 1.0 - outlier_penalty)

        # Staleness score (1.0 for fresh data, 0.0 for stale data)
        staleness = quality_checks.get('staleness', {})
        scores['staleness'] = 0.0 if staleness.get('is_stale', True) else 1.0

        # Consistency score
        consistency = quality_checks.get('consistency', {})
        scores['consistency'] = 1.0 if consistency.get('is_consistent', False) else 0.0

        # Duplicates score (inverse relationship with duplicate percentage)
        duplicates = quality_checks.get('duplicates', {})
        duplicate_penalty = duplicates.get('duplicate_percentage', 0.0)
        scores['duplicates'] = max(0.0, 1.0 - duplicate_penalty)

        # Calculate weighted average
        weighted_score = sum(
            weights.get(check, 0) * score
            for check, score in scores.items()
        )

        return min(1.0, max(0.0, weighted_score))

    def _quality_score_to_level(self, score: float) -> QualityLevel:
        """Convert quality score to quality level"""

        if score >= 0.95:
            return QualityLevel.EXCELLENT
        elif score >= 0.85:
            return QualityLevel.GOOD
        elif score >= 0.70:
            return QualityLevel.ACCEPTABLE
        elif score >= 0.50:
            return QualityLevel.POOR
        else:
            return QualityLevel.UNACCEPTABLE

    def _generate_recommendations(
        self,
        quality_checks: Dict[str, Any],
        source_type: str
    ) -> List[str]:
        """Generate data quality improvement recommendations"""

        recommendations = []

        # Completeness recommendations
        completeness = quality_checks.get('completeness', {})
        if not completeness.get('passes_threshold', True):
            recommendations.append(
                f"Improve {source_type} data completeness - currently {completeness.get('completeness_rate', 0):.2%}"
            )

        # Outlier recommendations
        outliers = quality_checks.get('outliers', {})
        if not outliers.get('passes_threshold', True):
            recommendations.append(
                f"Implement outlier detection for {source_type} - {outliers.get('outliers_detected', 0)} outliers detected"
            )

        # Staleness recommendations
        staleness = quality_checks.get('staleness', {})
        if staleness.get('is_stale', False):
            recommendations.append(
                f"Increase update frequency for {source_type} - data is {staleness.get('age_seconds', 0):.0f} seconds old"
            )

        # Consistency recommendations
        consistency = quality_checks.get('consistency', {})
        if not consistency.get('passes_threshold', True):
            recommendations.append(
                f"Fix timestamp consistency issues in {source_type} data"
            )

        # Duplicate recommendations
        duplicates = quality_checks.get('duplicates', {})
        if not duplicates.get('passes_threshold', True):
            recommendations.append(
                f"Remove duplicate data points from {source_type} - {duplicates.get('duplicates_found', 0)} duplicates found"
            )

        return recommendations

    def _log_quality_issues(self, validation_results: Dict[str, Any]) -> None:
        """Log quality issues for monitoring"""

        logger = logging.getLogger('data_quality')

        source_type = validation_results.get('source_type', 'unknown')
        quality_score = validation_results.get('quality_score', 0.0)
        issues = validation_results.get('issues', [])

        logger.warning(
            f"Data quality issues detected for {source_type}: "
            f"score={quality_score:.2f}, issues={issues}"
        )
```

**4. Enhanced Backtesting Framework (`src/backtesting/unified_backtest_engine.py`)**
```python
from typing import Dict, List, Any, Optional
import pandas as pd
import numpy as np
from datetime import datetime, timedelta

class UnifiedBacktestEngine:
    """Enhanced backtesting engine supporting combined price and non-price strategies"""

    def __init__(self):
        self.data_pipeline = UnifiedDataPipeline()
        self.cache_manager = UnifiedCacheManager()
        self.quality_validator = DataQualityValidator()

    async def run_unified_backtest(
        self,
        symbol: str,
        strategy_config: Dict[str, Any],
        start_date: datetime,
        end_date: datetime,
        data_sources: List[str] = ['price', 'hkma', 'sentiment']
    ) -> Dict[str, Any]:
        """Run backtest with combined price and non-price data"""

        # 1. Fetch unified data
        unified_data = await self.data_pipeline.fetch_unified_data(
            [symbol], start_date, end_date, data_sources
        )

        # 2. Validate data quality
        quality_results = {}
        for source, data_points in unified_data[symbol].items():
            quality_results[source] = await self.quality_validator.validate_unified_data(
                data_points, source
            )

        # 3. Prepare backtest dataframe
        backtest_df = self._prepare_backtest_dataframe(unified_data[symbol])

        # 4. Generate combined signals
        signals = await self._generate_combined_signals(
            backtest_df, strategy_config, quality_results
        )

        # 5. Execute backtest simulation
        backtest_results = await self._execute_backtest_simulation(
            backtest_df, signals, strategy_config
        )

        # 6. Calculate performance metrics
        performance_metrics = self._calculate_performance_metrics(
            backtest_results, strategy_config
        )

        # 7. Generate analysis report
        analysis_report = self._generate_analysis_report(
            performance_metrics, quality_results, strategy_config
        )

        return {
            'symbol': symbol,
            'period': f"{start_date.date()} to {end_date.date()}",
            'data_sources': data_sources,
            'quality_results': quality_results,
            'performance_metrics': performance_metrics,
            'analysis_report': analysis_report,
            'raw_results': backtest_results
        }

    def _prepare_backtest_dataframe(
        self,
        unified_data: List[UnifiedDataPoint]
    ) -> pd.DataFrame:
        """Convert unified data points to backtest dataframe"""

        # Group by timestamp
        timestamp_groups = {}
        for data_point in unified_data:
            if data_point.timestamp not in timestamp_groups:
                timestamp_groups[data_point.timestamp] = {}

            timestamp_groups[data_point.timestamp][data_point.source] = data_point

        # Create dataframe
        rows = []
        for timestamp in sorted(timestamp_groups.keys()):
            row = {'timestamp': timestamp}

            # Add price data
            if 'price' in timestamp_groups['timestamp']:
                price_data = timestamp_groups['timestamp']['price']
                row['close'] = price_data.value
                row['volume'] = price_data.metadata.get('volume', 0)

            # Add HKMA data
            hkma_data = timestamp_groups['timestamp'].get('hkma', {})
            row['hibor'] = hkma_data.get('hibor', {}).get('value', np.nan)
            row['monetary_base'] = hkma_data.get('monetary_base', {}).get('value', np.nan)

            # Add sentiment data
            sentiment_data = timestamp_groups['timestamp'].get('sentiment', {})
            row['sentiment'] = sentiment_data.get('value', np.nan)
            row['sentiment_confidence'] = sentiment_data.get('confidence', np.nan)

            rows.append(row)

        df = pd.DataFrame(rows)
        df.set_index('timestamp', inplace=True)

        # Forward fill missing values
        df = df.fillna(method='ffill')
        df = df.fillna(method='bfill')

        return df

    async def _generate_combined_signals(
        self,
        df: pd.DataFrame,
        strategy_config: Dict[str, Any],
        quality_results: Dict[str, Any]
    ) -> pd.DataFrame:
        """Generate combined signals from price and non-price data"""

        signals = pd.DataFrame(index=df.index)

        # Generate price-based signals
        if 'price' in quality_results and quality_results['price']['quality_score'] > 0.7:
            signals['price_signal'] = self._generate_price_signals(df, strategy_config.get('price', {}))

        # Generate non-price signals
        if 'hkma' in quality_results and quality_results['hkma']['quality_score'] > 0.7:
            signals['hkma_signal'] = self._generate_hkma_signals(df, strategy_config.get('hkma', {}))

        if 'sentiment' in quality_results and quality_results['sentiment']['quality_score'] > 0.7:
            signals['sentiment_signal'] = self._generate_sentiment_signals(df, strategy_config.get('sentiment', {}))

        # Combine signals using configured weights
        signal_weights = strategy_config.get('signal_weights', {
            'price': 0.5,
            'hkma': 0.3,
            'sentiment': 0.2
        })

        combined_signal = pd.Series(0.0, index=df.index)
        total_weight = 0.0

        for signal_type, weight in signal_weights.items():
            signal_col = f"{signal_type}_signal"
            if signal_col in signals:
                combined_signal += signals[signal_col] * weight
                total_weight += weight

        if total_weight > 0:
            combined_signal = combined_signal / total_weight

        signals['combined_signal'] = combined_signal
        signals['final_signal'] = (combined_signal > 0).astype(int)  # Binary signal

        return signals

    def _generate_hkma_signals(self, df: pd.DataFrame, config: Dict[str, Any]) -> pd.Series:
        """Generate trading signals from HKMA macro data"""

        signals = pd.Series(0, index=df.index)

        # HIBOR-based signals
        if 'hibor' in df.columns:
            hibor_ma = df['hibor'].rolling(window=config.get('hibor_ma', 20)).mean()
            signals.loc[df['hibor'] > hibor_ma * 1.05] = 1  # High rates = sell signal
            signals.loc[df['hibor'] < hibor_ma * 0.95] = -1  # Low rates = buy signal

        # Monetary base signals
        if 'monetary_base' in df.columns:
            mb_ma = df['monetary_base'].rolling(window=config.get('mb_ma', 30)).mean()
            mb_change = df['monetary_base'].pct_change()

            # Expanding monetary base = bullish
            signals.loc[(df['monetary_base'] > mb_ma) & (mb_change > 0.01)] = 1
            # Contracting monetary base = bearish
            signals.loc[(df['monetary_base'] < mb_ma) & (mb_change < -0.01)] = -1

        return signals

    def _generate_sentiment_signals(self, df: pd.DataFrame, config: Dict[str, Any]) -> pd.Series:
        """Generate trading signals from sentiment data"""

        signals = pd.Series(0, index=df.index)

        if 'sentiment' in df.columns:
            sentiment_ma = df['sentiment'].rolling(window=config.get('sentiment_ma', 5)).mean()

            # Strong bullish sentiment = buy signal
            signals.loc[df['sentiment'] > sentiment_ma + config.get('sentiment_threshold', 0.2)] = 1
            # Strong bearish sentiment = sell signal
            signals.loc[df['sentiment'] < sentiment_ma - config.get('sentiment_threshold', 0.2)] = -1

            # Consider sentiment confidence
            if 'sentiment_confidence' in df.columns:
                confidence_mask = df['sentiment_confidence'] > config.get('min_confidence', 0.7)
                signals = signals * confidence_mask

        return signals

    def _calculate_performance_metrics(
        self,
        backtest_results: Dict[str, Any],
        strategy_config: Dict[str, Any]
    ) -> Dict[str, Any]:
        """Calculate comprehensive performance metrics"""

        returns = backtest_results['returns']
        equity_curve = backtest_results['equity_curve']

        # Basic metrics
        total_return = (equity_curve[-1] / equity_curve[0]) - 1
        annualized_return = (1 + total_return) ** (252 / len(equity_curve)) - 1

        # Risk metrics
        returns_array = np.array(returns)
        volatility = np.std(returns_array) * np.sqrt(252)
        downside_volatility = np.std(returns_array[returns_array < 0]) * np.sqrt(252)

        # Sharpe and Sortino ratios
        sharpe_ratio = annualized_return / volatility if volatility > 0 else 0
        sortino_ratio = annualized_return / downside_volatility if downside_volatility > 0 else 0

        # Maximum drawdown
        equity_array = np.array(equity_curve)
        running_max = np.maximum.accumulate(equity_array)
        drawdowns = (equity_array - running_max) / running_max
        max_drawdown = np.min(drawdowns)

        # Win rate and profit factor
        winning_trades = np.sum(returns_array > 0)
        losing_trades = np.sum(returns_array < 0)
        win_rate = winning_trades / (winning_trades + losing_trades) if (winning_trades + losing_trades) > 0 else 0

        gross_profit = np.sum(returns_array[returns_array > 0])
        gross_loss = abs(np.sum(returns_array[returns_array < 0]))
        profit_factor = gross_profit / gross_loss if gross_loss > 0 else float('inf')

        return {
            'total_return': total_return,
            'annualized_return': annualized_return,
            'volatility': volatility,
            'sharpe_ratio': sharpe_ratio,
            'sortino_ratio': sortino_ratio,
            'max_drawdown': max_drawdown,
            'win_rate': win_rate,
            'profit_factor': profit_factor,
            'total_trades': len(returns_array),
            'winning_trades': int(winning_trades),
            'losing_trades': int(losing_trades)
        }
```
