**2. Data Integration Layer (`src/services/nonPriceService.ts`)**
```typescript
import { apiClient } from './apiClient';

interface NonPriceSignal {
  id: string;
  type: 'macro' | 'sentiment' | 'alternative';
  source: string;
  timestamp: Date;
  value: number;
  confidence: number;
  metadata: Record<string, any>;
}

class NonPriceService {
  private wsConnections: Map<string, WebSocket> = new Map();

  async getMacroIndicators(indicatorType: string): Promise<MacroIndicator[]> {
    const response = await apiClient.get(`/api/non-price/hkma/${indicatorType}/latest`);
    return response.data;
  }

  async getSentimentData(symbol: string): Promise<SentimentData> {
    const response = await apiClient.get(`/api/non-price/sentiment/latest/${symbol}`);
    return response.data;
  }

  subscribeToRealTimeSignals(symbol: string, callback: (signal: NonPriceSignal) => void): void {
    const ws = new WebSocket(`ws://localhost:3004/api/non-price/ws/signals/${symbol}`);
    ws.onmessage = (event) => {
      const signal = JSON.parse(event.data);
      callback(signal);
    };
    this.wsConnections.set(symbol, ws);
  }
}

export const nonPriceService = new NonPriceService();
```

```typescript
import React, { useState, useEffect } from 'react';
import { LineChart, Line, XAxis, YAxis, CartesianGrid, Tooltip, ResponsiveContainer } from 'recharts';
import { nonPriceService } from '../../services/nonPriceService';

interface HIBORData {
  timestamp: Date;
  value: number;
  trend: 'UP' | 'DOWN' | 'STABLE';
}

const HIBORDisplay: React.FC = () => {
  const [hiborData, setHiborData] = useState<HIBORData[]>([]);
  const [currentRate, setCurrentRate] = useState<number>(0);
  const [loading, setLoading] = useState<boolean>(true);

  useEffect(() => {
    const fetchHIBORData = async () => {
      try {
        const data = await nonPriceService.getMacroIndicators('hibor');
        setHiborData(data.historical);
        setCurrentRate(data.current);
      } catch (error) {
        console.error('Failed to fetch HIBOR data:', error);
      } finally {
        setLoading(false);
      }
    };

    fetchHIBORData();
    const interval = setInterval(fetchHIBORData, 60000); // Update every minute

    return () => clearInterval(interval);
  }, []);

  const getTrendColor = (trend: string) => {
    switch (trend) {
      case 'UP': return '#ef4444';
      case 'DOWN': return '#22c55e';
      default: return '#3b82f6';
    }
  };

  if (loading) {
    return <div className="p-4">Loading HIBOR data...</div>;
  }

  return (
    <div className="bg-white rounded-lg shadow-md p-6">
      <div className="flex justify-between items-center mb-4">
        <h3 className="text-lg font-semibold">HIBOR Rate</h3>
        <div className="text-2xl font-bold">
          {currentRate.toFixed(2)}%
        </div>
      </div>

      <ResponsiveContainer width="100%" height={200}>
        <LineChart data={hiborData}>
          <CartesianGrid strokeDasharray="3 3" />
          <XAxis
            dataKey="timestamp"
            tickFormatter={(value) => new Date(value).toLocaleTimeString()}
          />
          <YAxis domain={['dataMin - 0.1', 'dataMax + 0.1']} />
          <Tooltip
            labelFormatter={(value) => new Date(value).toLocaleString()}
            formatter={(value: any) => [`${value}%`, 'HIBOR']}
          />
          <Line
            type="monotone"
            dataKey="value"
            stroke="#3b82f6"
            strokeWidth={2}
            dot={false}
          />
        </LineChart>
      </ResponsiveContainer>

      <div className="mt-4 text-sm text-gray-600">
        <p>Real-time Hong Kong Interbank Offered Rate</p>
        <p>Last updated: {new Date().toLocaleTimeString()}</p>
      </div>
    </div>
  );
};

export default HIBORDisplay;
```

**SentimentAnalysis/SentimentGauge.tsx**
```typescript
import React from 'react';

interface SentimentGaugeProps {
  sentiment: number; // -1 to 1 scale
  confidence: number; // 0 to 1 scale
  symbol: string;
}

const SentimentGauge: React.FC<SentimentGaugeProps> = ({ sentiment, confidence, symbol }) => {
  const getSentimentColor = (value: number) => {
    if (value > 0.3) return '#22c55e'; // Bullish - Green
    if (value < -0.3) return '#ef4444'; // Bearish - Red
    return '#f59e0b'; // Neutral - Amber
  };

  const getSentimentLabel = (value: number) => {
    if (value > 0.5) return 'Strongly Bullish';
    if (value > 0.3) return 'Bullish';
    if (value > -0.3) return 'Neutral';
    if (value > -0.5) return 'Bearish';
    return 'Strongly Bearish';
  };

  const angle = (sentiment + 1) * 90; // Convert -1 to 1 range to 0 to 180 degrees

  return (
    <div className="bg-white rounded-lg shadow-md p-6">
      <h3 className="text-lg font-semibold mb-4">Market Sentiment - {symbol}</h3>

      <div className="relative w-48 h-24 mx-auto">
        <div className="absolute inset-0 bg-gray-200 rounded-full"></div>
        <div
          className="absolute inset-0 rounded-full transition-all duration-500"
          style={{
            background: `conic-gradient(from 180deg, ${getSentimentColor(sentiment)} 0deg, ${getSentimentColor(sentiment)} ${angle}deg, #e5e7eb ${angle}deg)`
          }}
        ></div>
        <div className="absolute inset-2 bg-white rounded-full flex items-center justify-center">
          <div className="text-center">
            <div className="text-2xl font-bold" style={{ color: getSentimentColor(sentiment) }}>
              {((sentiment + 1) * 50).toFixed(0)}
            </div>
            <div className="text-xs text-gray-500">Score</div>
          </div>
        </div>
      </div>

      <div className="mt-4 text-center">
        <div className="text-lg font-medium" style={{ color: getSentimentColor(sentiment) }}>
          {getSentimentLabel(sentiment)}
        </div>
        <div className="text-sm text-gray-600">
          Confidence: {(confidence * 100).toFixed(1)}%
        </div>
      </div>

      <div className="mt-4 p-3 bg-gray-50 rounded">
        <div className="text-xs text-gray-600">
          <p>Based on social media sentiment analysis</p>
          <p>News sentiment: {(sentiment * 0.6 + confidence * 0.4).toFixed(2)}</p>
          <p>Technical sentiment: {(sentiment * 0.8).toFixed(2)}</p>
        </div>
      </div>
    </div>
  );
};

export default SentimentGauge;
```

**RealTimeSignalUpdater.tsx**
```typescript
import React, { useEffect, useState, useCallback } from 'react';
import { nonPriceService } from '../../services/nonPriceService';

interface UseRealTimeSignalsResult {
  signals: NonPriceSignal[];
  isConnected: boolean;
  lastUpdate: Date | null;
}

export const useRealTimeSignals = (symbol: string): UseRealTimeSignalsResult => {
  const [signals, setSignals] = useState<NonPriceSignal[]>([]);
  const [isConnected, setIsConnected] = useState<boolean>(false);
  const [lastUpdate, setLastUpdate] = useState<Date | null>(null);

  useEffect(() => {
    let ws: WebSocket | null = null;

    const connectWebSocket = () => {
      try {
        ws = new WebSocket(`ws://localhost:3004/api/non-price/ws/signals/${symbol}`);

        ws.onopen = () => {
          setIsConnected(true);
          console.log(`Connected to non-price signals for ${symbol}`);
        };

        ws.onmessage = (event) => {
          const signal: NonPriceSignal = JSON.parse(event.data);
          setSignals(prev => [...prev.slice(-99), signal]); // Keep last 100 signals
          setLastUpdate(new Date());
        };

        ws.onclose = () => {
          setIsConnected(false);
          console.log(`Disconnected from non-price signals for ${symbol}`);
          // Attempt to reconnect after 5 seconds
          setTimeout(connectWebSocket, 5000);
        };

        ws.onerror = (error) => {
          console.error('WebSocket error:', error);
          setIsConnected(false);
        };

      } catch (error) {
        console.error('Failed to connect to WebSocket:', error);
        setIsConnected(false);
      }
    };

    connectWebSocket();

    return () => {
      if (ws) {
        ws.close();
      }
    };
  }, [symbol]);

  return { signals, isConnected, lastUpdate };
};
```

### User Preferences

**NonPriceSettings.tsx**
```typescript
import React, { useState, useEffect } from 'react';

interface UserNonPricePreferences {
  enabledStrategies: string[];
  refreshInterval: number;
  alertThreshold: number;
  defaultTimeframe: string;
  showAdvancedMetrics: boolean;
}

const NonPriceSettings: React.FC = () => {
  const [preferences, setPreferences] = useState<UserNonPricePreferences>({
    enabledStrategies: ['hibor', 'sentiment', 'monetary_base'],
    refreshInterval: 60,
    alertThreshold: 0.7,
    defaultTimeframe: '1D',
    showAdvancedMetrics: false
  });

  const savePreferences = async () => {
    try {
      await fetch('/api/user/preferences/non-price', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(preferences)
      });
    } catch (error) {
      console.error('Failed to save preferences:', error);
    }
  };

  return (
    <div className="bg-white rounded-lg shadow-md p-6">
      <h3 className="text-lg font-semibold mb-4">Non-Price Strategy Settings</h3>

      <div className="space-y-4">
        <div>
          <label className="block text-sm font-medium text-gray-700 mb-2">
            Enabled Strategies
          </label>
          <div className="space-y-2">
            {['hibor', 'sentiment', 'monetary_base', 'exchange_rate'].map(strategy => (
              <label key={strategy} className="flex items-center">
                <input
                  type="checkbox"
                  checked={preferences.enabledStrategies.includes(strategy)}
                  onChange={(e) => {
                    const updated = e.target.checked
                      ? [...preferences.enabledStrategies, strategy]
                      : preferences.enabledStrategies.filter(s => s !== strategy);
                    setPreferences({ ...preferences, enabledStrategies: updated });
                  }}
                  className="mr-2"
                />
                <span className="capitalize">{strategy.replace('_', ' ')}</span>
              </label>
            ))}
          </div>
        </div>

        <div>
          <label className="block text-sm font-medium text-gray-700 mb-2">
            Refresh Interval (seconds)
          </label>
          <input
            type="number"
            value={preferences.refreshInterval}
            onChange={(e) => setPreferences({ ...preferences, refreshInterval: parseInt(e.target.value) })}
            className="w-full px-3 py-2 border border-gray-300 rounded-md"
            min="10"
            max="300"
          />
        </div>

        <div>
          <label className="block text-sm font-medium text-gray-700 mb-2">
            Alert Confidence Threshold
          </label>
          <input
            type="range"
            min="0"
            max="1"
            step="0.1"
            value={preferences.alertThreshold}
            onChange={(e) => setPreferences({ ...preferences, alertThreshold: parseFloat(e.target.value) })}
            className="w-full"
          />
          <div className="text-sm text-gray-600 text-center">
            {(preferences.alertThreshold * 100).toFixed(0)}%
          </div>
        </div>

        <button
          onClick={savePreferences}
          className="w-full bg-blue-600 text-white px-4 py-2 rounded-md hover:bg-blue-700"
        >
          Save Preferences
        </button>
      </div>
    </div>
  );
};

export default NonPriceSettings;
```
