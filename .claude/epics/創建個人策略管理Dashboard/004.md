### 圖表組件設計

#### Sharpe比率條形圖 (SharpeRatioChart)
```javascript
class SharpeRatioChart {
  constructor(canvasElement) {
    this.canvas = canvasElement;
    this.chart = null;
  }

  createChart(data) {
    const ctx = this.canvas.getContext('2d');

    this.chart = new Chart(ctx, {
      type: 'bar',
      data: {
        labels: data.map(s => s.name),
        datasets: [{
          label: 'Sharpe比率',
          data: data.map(s => s.sharpeRatio),
          backgroundColor: data.map(s =>
            s.sharpeRatio >= 1 ? chartTheme.success :
            s.sharpeRatio >= 0.5 ? chartTheme.warning : chartTheme.danger
          ),
          borderColor: chartTheme.dark,
          borderWidth: 1
        }]
      },
      options: {
        responsive: true,
        plugins: {
          title: {
            display: true,
            text: '策略Sharpe比率排名'
          },
          tooltip: {
            callbacks: {
              label: function(context) {
                return `${context.parsed.y.toFixed(2)}`;
              }
            }
          }
        },
        scales: {
          y: {
            beginAtZero: true,
            title: {
              display: true,
              text: 'Sharpe比率'
            }
          }
        }
      }
    });
  }

  updateChart(newData) { /* 更新圖表數據 */ }
  destroy() { /* 銷毀圖表 */ }
}
```

#### 最大回撤折線圖 (MaxDrawdownChart)
```javascript
class MaxDrawdownChart {
  constructor(canvasElement) {
    this.canvas = canvasElement;
    this.chart = null;
  }

  createChart(historicalData) {
    const ctx = this.canvas.getContext('2d');

    this.chart = new Chart(ctx, {
      type: 'line',
      data: {
        labels: historicalData.dates,
        datasets: historicalData.strategies.map(strategy => ({
          label: strategy.name,
          data: strategy.drawdowns,
          borderColor: strategyColors[strategy.name],
          backgroundColor: strategyColors[strategy.name] + '20',
          tension: 0.4,
          fill: false
        }))
      },
      options: {
        responsive: true,
        plugins: {
          title: {
            display: true,
            text: '策略最大回撤趨勢'
          }
        },
        scales: {
          y: {
            beginAtZero: true,
            title: {
              display: true,
              text: '最大回撤 (%)'
            }
          }
        }
      }
    });
  }
}
```

#### 策略雷達圖 (StrategyRadarChart)
```javascript
class StrategyRadarChart {
  constructor(canvasElement) {
    this.canvas = canvasElement;
    this.chart = null;
  }

  createChart(multiDimensionData) {
    const ctx = this.canvas.getContext('2d');

    this.chart = new Chart(ctx, {
      type: 'radar',
      data: {
        labels: ['Sharpe比率', '最大回撤', '勝率', '夏普比率', '波動率控制'],
        datasets: multiDimensionData.map(strategy => ({
          label: strategy.name,
          data: [
            strategy.sharpeRatio * 20,
            (1 - strategy.maxDrawdown) * 100,
            strategy.winRate * 100,
            strategy.sharpeRatio * 25,
            strategy.volatilityControl * 100
          ],
          borderColor: strategyColors[strategy.name],
          backgroundColor: strategyColors[strategy.name] + '30'
        }))
      },
      options: {
        responsive: true,
        plugins: {
          title: {
            display: true,
            text: '策略多維度性能對比'
          }
        },
        scales: {
          r: {
            beginAtZero: true,
            max: 100
          }
        }
      }
    });
  }
}
```
