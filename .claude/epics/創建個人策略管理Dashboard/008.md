#### 性能測試
```javascript
// 性能測試工具
class PerformanceTester {
  constructor() {
    this.metrics = {};
  }

  // 頁面加載性能測試
  async testPageLoadPerformance() {
    const startTime = performance.now();

    // 模擬頁面加載
    window.location.reload();

    window.addEventListener('load', () => {
      const loadTime = performance.now() - startTime;
      this.metrics.pageLoadTime = loadTime;

      console.log(`Page load time: ${loadTime}ms`);

      // 驗證性能指標
      this.validatePerformanceMetrics();
    });
  }

  // API響應時間測試
  async testAPIResponseTime() {
    const apiEndpoints = [
      '/api/strategies/performance',
      '/api/strategies/list'
    ];

    for (const endpoint of apiEndpoints) {
      const startTime = performance.now();

      try {
        const response = await fetch(endpoint);
        const responseTime = performance.now() - startTime;

        this.metrics[endpoint] = responseTime;
        console.log(`API ${endpoint} response time: ${responseTime}ms`);

        // 驗證響應時間是否 < 500ms
        if (responseTime > 500) {
          console.warn(`Slow API response detected: ${endpoint} - ${responseTime}ms`);
        }
      } catch (error) {
        console.error(`API call failed: ${endpoint} - ${error.message}`);
      }
    }
  }

  // 圖表渲染性能測試
  testChartRenderingPerformance() {
    const charts = document.querySelectorAll('canvas');

    charts.forEach((chart, index) => {
      const startTime = performance.now();

      // 模擬圖表渲染完成
      setTimeout(() => {
        const renderTime = performance.now() - startTime;
        this.metrics[`chart_${index}`] = renderTime;

        console.log(`Chart ${index} render time: ${renderTime}ms`);

        // 驗證渲染時間是否 < 1秒
        if (renderTime > 1000) {
          console.warn(`Slow chart rendering detected: Chart ${index} - ${renderTime}ms`);
        }
      }, 100);
    });
  }

  validatePerformanceMetrics() {
    const targets = {
      pageLoadTime: 2000,      // 2秒
      apiResponse: 500,       // 500ms
      chartRender: 1000        // 1秒
    };

    let allTestsPassed = true;

    Object.entries(this.metrics).forEach(([metric, value]) => {
      const target = targets[metric];
      if (value > target) {
        console.error(`Performance test failed: ${metric} - ${value}ms (target: ${target}ms)`);
        allTestsPassed = false;
      } else {
        console.log(`Performance test passed: ${metric} - ${value}ms`);
      }
    });

    return allTestsPassed;
  }
}
```

### 部署配置

#### 靜態文件部署
```bash
# 創建部署腳本
#!/bin/bash
echo "部署個人策略管理Dashboard..."

# 創建目錄結構
mkdir -p strategy-dashboard/{css,js,assets}

# 複製文件
cp index.html strategy-dashboard/
cp css/*.css strategy-dashboard/css/
cp js/*.js strategy-dashboard/js/
cp assets/chart.js strategy-dashboard/assets/

# 配置FastAPI靜態文件服務
python -c "
from fastapi import FastAPI
from fastapi.staticfiles import StaticFiles

app = FastAPI()

# 添加靜態文件服務
app.mount('/dashboard', StaticFiles(directory='strategy-dashboard', html=True))

print('Dashboard部署完成！')
print('訪問地址: http://localhost:3003/dashboard')
"
```
