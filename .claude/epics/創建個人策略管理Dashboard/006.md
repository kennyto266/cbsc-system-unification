### WebSocket實現
```javascript
class WebSocketClient {
  constructor(url) {
    this.url = url;
    this.ws = null;
    this.reconnectAttempts = 0;
    this.maxReconnectAttempts = 5;
    this.reconnectDelay = 1000;
  }

  connect() {
    try {
      this.ws = new WebSocket(this.url);
      this.setupEventHandlers();
    } catch (error) {
      console.error('WebSocket連接失敗:', error);
      this.handleReconnect();
    }
  }

  setupEventHandlers() {
    this.ws.onopen = (event) => {
      console.log('WebSocket連接已建立');
      this.reconnectAttempts = 0;
      this.send('subscribe', { channels: ['strategy_performance'] });
    };

    this.ws.onmessage = (event) => {
      const data = JSON.parse(event.data);
      this.handleRealtimeData(data);
    };

    this.ws.onclose = (event) => {
      console.log('WebSocket連接已關閉');
      this.handleReconnect();
    };

    this.ws.onerror = (error) => {
      console.error('WebSocket錯誤:', error);
    };
  }

  handleReconnect() {
    if (this.reconnectAttempts < this.maxReconnectAttempts) {
      setTimeout(() => {
        this.reconnectAttempts++;
        this.connect();
      }, this.reconnectDelay * Math.pow(2, this.reconnectAttempts));
    }
  }
}
```

### 定時器管理系統
```javascript
class TimerManager {
  constructor(interval = 10000) {
    this.interval = interval;
    this.timer = null;
    this.isActive = false;
    this.callbacks = [];
  }

  start() {
    if (!this.isActive) {
      this.isActive = true;
      this.timer = setInterval(() => {
        this.executeCallbacks();
      }, this.interval);
    }
  }

  stop() {
    if (this.isActive) {
      this.isActive = false;
      clearInterval(this.timer);
      this.timer = null;
    }
  }

  restart() {
    this.stop();
    this.start();
  }

  addCallback(callback) {
    this.callbacks.push(callback);
  }

  executeCallbacks() {
    this.callbacks.forEach(callback => {
      try {
        callback();
      } catch (error) {
        console.error('定時器回調執行錯誤:', error);
      }
    });
  }
}
```

### 數據同步邏輯
```javascript
class DataSynchronizer {
  constructor(apiClient, chartManager) {
    this.apiClient = apiClient;
    this.chartManager = chartManager;
    this.lastDataHash = null;
    this.updateQueue = [];
  }

  async syncData() {
    try {
      this.showLoadingIndicator();

      const [performanceData, strategyList] = await Promise.all([
        this.apiClient.getStrategyPerformance(),
        this.apiClient.getStrategyList()
      ]);

      const newDataHash = this.calculateDataHash(performanceData, strategyList);

      if (newDataHash !== this.lastDataHash) {
        this.updateUI(performanceData, strategyList);
        this.lastDataHash = newDataHash;
      }

      this.hideLoadingIndicator();
      this.updateLastSyncTime();

    } catch (error) {
      this.handleSyncError(error);
    }
  }

  calculateDataHash(performanceData, strategyList) {
    const dataString = JSON.stringify({ performanceData, strategyList });
    return this.simpleHash(dataString);
  }

  simpleHash(str) {
    let hash = 0;
    for (let i = 0; i < str.length; i++) {
      const char = str.charCodeAt(i);
      hash = ((hash << 5) - hash) + char;
      hash = hash & hash;
    }
    return hash.toString();
  }
}
```

```javascript
class NetworkMonitor {
  constructor() {
    this.isOnline = navigator.onLine;
    this.setupEventListeners();
  }

  setupEventListeners() {
    window.addEventListener('online', () => {
      this.isOnline = true;
      this.showNetworkStatus('online');
      this.triggerResync();
    });

    window.addEventListener('offline', () => {
      this.isOnline = false;
      this.showNetworkStatus('offline');
    });
  }

  showNetworkStatus(status) {
    const statusElement = document.getElementById('network-status');
    if (statusElement) {
      statusElement.className = status;
      statusElement.textContent = status === 'online' ? '已連接' : '離線';
    }
  }

  triggerResync() {
    if (this.onOnlineCallback) {
      this.onOnlineCallback();
    }
  }
}
```

```javascript
        this.hideChartLoading();

      this.overlayElement.style.display = 'flex';
    }

    }
  }
}
```

### 用戶交互界面
```javascript
class RefreshControl {
  constructor() {
    this.refreshButton = document.getElementById('refresh-button');
    this.lastSyncElement = document.getElementById('last-sync-time');
    this.setupEventListeners();
  }

  setupEventListeners() {
    this.refreshButton.addEventListener('click', () => {
      this.triggerManualRefresh();
    });
  }

  triggerManualRefresh() {
    this.refreshButton.disabled = true;
    this.refreshButton.classList.add('loading');

    // 觸發數據同步
    if (this.onRefreshCallback) {
      this.onRefreshCallback().finally(() => {
        this.refreshButton.disabled = false;
        this.refreshButton.classList.remove('loading');
      });
    }
  }

  updateLastSyncTime() {
    if (this.lastSyncElement) {
      const now = new Date();
      this.lastSyncElement.textContent = `最後更新: ${now.toLocaleTimeString()}`;
    }
  }
}
```
