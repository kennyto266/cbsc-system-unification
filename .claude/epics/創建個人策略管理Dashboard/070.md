### 测试工具函数

```typescript
// tests/utils/test-helpers.ts
export async function loginAsTestUser(page: Page) {
  await page.goto('/login');
  await page.fill('[data-testid=username]', process.env.TEST_USER || 'testuser');
  await page.fill('[data-testid=password]', process.env.TEST_PASS || 'testpass123');
  await page.click('[data-testid=login-button]');
  await page.waitForURL('/dashboard');
}

export async function createTestStrategy(page: Page, strategyData: Partial<Strategy>) {
  await page.click('[data-testid=create-strategy-btn]');
  await page.fill('[data-testid=strategy-name]', strategyData.name || 'Auto Test Strategy');
  await page.selectOption('[data-testid=strategy-type]', strategyData.type || 'momentum');
  await page.click('[data-testid=save-strategy-btn]');
  await expect(page.locator('[data-testid=success-message]')).toBeVisible();
}

export async function triggerStrategyExecution(strategyId: string) {
  // 通过API触发策略执行
  await fetch(`${process.env.API_URL}/api/strategies/${strategyId}/execute`, {
    method: 'POST',
    headers: {
      'Authorization': `Bearer ${process.env.TEST_TOKEN}`,
      'Content-Type': 'application/json'
    }
  });
}
```

### 数据工厂

```typescript
// tests/factories/strategy.factory.ts
export class StrategyFactory {
  static create(overrides: Partial<Strategy> = {}): Strategy {
    return {
      id: faker.datatype.uuid(),
      name: faker.company.catchPhrase(),
      type: faker.helpers.arrayElement(['momentum', 'mean_reversion', 'trend_following']),
      status: 'inactive',
      initial_capital: 100000,
      created_at: faker.date.recent(),
      ...overrides
    };
  }

  static createMany(count: number, overrides: Partial<Strategy> = {}): Strategy[] {
    return Array.from({ length: count }, () => this.create(overrides));
  }
}
```

### CI/CD集成

```yaml
# .github/workflows/e2e-tests.yml
name: E2E Tests

on:
  pull_request:
    branches: [main]
  schedule:
    - cron: '0 2 * * *'  # 每天凌晨2点运行

jobs:
  e2e-tests:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3

      - name: Setup Node.js
        uses: actions/setup-node@v3
        with:
          node-version: '18'
          cache: 'npm'

      - name: Install dependencies
        run: npm ci

      - name: Install Playwright Browsers
        run: npx playwright install --with-deps

      - name: Start test server
        run: npm run test:start &

      - name: Run Playwright tests
        run: npx playwright test

      - name: Upload test results
        uses: actions/upload-artifact@v3
        if: failure()
        with:
          name: playwright-report
          path: playwright-report/
```
