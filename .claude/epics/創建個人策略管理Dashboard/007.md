### API接口調用
```javascript
// 策略切換API
async function toggleStrategy(strategyName, enabled) {
  try {
    const response = await fetch(`/api/strategies/${strategyName}/toggle`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({ enabled: enabled })
    });

    if (response.ok) {
      const result = await response.json();
      return { success: true, data: result };
    } else {
      throw new Error(`HTTP ${response.status}: ${response.statusText}`);
    }
  } catch (error) {
    return { success: false, error: error.message };
  }
}
```

### 狀態管理
```javascript
// 策略狀態管理
class StrategyStateManager {
  constructor() {
    this.strategies = {
      DirectRSIStrategy: false,
      SentimentMomentumStrategy: false,
      CompositeIndexStrategy: false,
      VolatilityAdjustedStrategy: false
    };
    this.listeners = [];
  }

  // 切換策略狀態
  async toggleStrategy(strategyName) {
    const currentState = this.strategies[strategyName];
    const newState = !currentState;

    // 顯示確認對話框
    const confirmed = await this.showConfirmDialog(
      `確定要${newState ? '啟用' : '禁用'}策略 "${strategyName}"嗎？`
    );

    if (!confirmed) return false;

    // 調用API
    const result = await toggleStrategy(strategyName, newState);

    if (result.success) {
      this.strategies[strategyName] = newState;
      this.notifyListeners(strategyName, newState);
      return true;
    } else {
      this.showError(`切換策略失敗: ${result.error}`);
      return false;
    }
  }

  // 批量切換
  async toggleAllStrategies(enabled) {
    const confirmed = await this.showConfirmDialog(
      `確定要${enabled ? '啟用' : '禁用'}所有策略嗎？`
    );

    if (!confirmed) return;

    const results = [];
    for (const strategyName in this.strategies) {
      if (this.strategies[strategyName] !== enabled) {
        const result = await this.toggleStrategy(strategyName, false); // 不需要確認對話框
        results.push({ strategyName, result });
      }
    }

    const successCount = results.filter(r => r.result).length;
    const totalCount = results.length;

    this.showNotification(
      `批量操作完成：${successCount}/${totalCount} 個策略切換成功`,
      successCount === totalCount ? 'success' : 'warning'
    );
  }
}
```
