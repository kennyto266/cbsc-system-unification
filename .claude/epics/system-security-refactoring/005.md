### 1. Redis分布式缓存架构

```python
# Redis缓存配置和连接管理
import redis
from redis.sentinel import Sentinel
from typing import Any, Optional
import json
import pickle
from functools import wraps

class RedisCacheManager:
    """Redis分布式缓存管理器"""

    def __init__(self, config: dict):
        self.config = config
        self.sentinel = None
        self.master = None
        self.slave = None
        self.connection_pool = None
        self._initialize_connections()

    def _initialize_connections(self):
        """初始化Redis连接"""
        # 主从复制配置
        self.sentinel = Sentinel([
            (self.config['sentinel_host'], self.config['sentinel_port'])
        ])

        # 获取主从连接
        self.master = self.sentinel.master_for(
            self.config['master_name'],
            socket_timeout=self.config['timeout'],
            socket_connect_timeout=self.config['connect_timeout'],
            retry_on_timeout=True
        )

        self.slave = self.sentinel.slave_for(
            self.config['master_name'],
            socket_timeout=self.config['timeout'],
            socket_connect_timeout=self.config['connect_timeout']
        )

        # 连接池配置
        self.connection_pool = redis.ConnectionPool(
            host=self.config['host'],
            port=self.config['port'],
            max_connections=self.config['max_connections'],
            retry_on_timeout=True
        )

    def get(self, key: str) -> Optional[Any]:
        """获取缓存数据"""
        try:
            # 优先从从库读取
            data = self.slave.get(key)
            if data:
                return pickle.loads(data)
        except Exception as e:
            # 从库故障时从主库读取
            try:
                data = self.master.get(key)
                if data:
                    return pickle.loads(data)
            except Exception:
                pass
        return None

    def set(self, key: str, value: Any, ttl: int = 3600) -> bool:
        """设置缓存数据"""
        try:
            serialized_data = pickle.dumps(value)
            return self.master.setex(key, ttl, serialized_data)
        except Exception:
            return False

    def delete(self, key: str) -> bool:
        """删除缓存数据"""
        try:
            return bool(self.master.delete(key))
        except Exception:
            return False
```

### 2. 本地内存缓存实现

```python
import time
from threading import RLock
from collections import OrderedDict
from typing import Any, Optional

class LRUCache:
    """LRU本地内存缓存"""

    def __init__(self, max_size: int = 1000, max_memory: int = 100 * 1024 * 1024):
        self.max_size = max_size
        self.max_memory = max_memory
        self.current_memory = 0
        self.cache = OrderedDict()
        self.lock = RLock()
        self.stats = {
            'hits': 0,
            'misses': 0,
            'evictions': 0
        }

    def get(self, key: str) -> Optional[Any]:
        """获取缓存项"""
        with self.lock:
            if key in self.cache:
                value, expire_time = self.cache[key]

                # 检查是否过期
                if expire_time and time.time() > expire_time:
                    self._remove_item(key)
                    self.stats['misses'] += 1
                    return None

                # 移动到末尾（最近使用）
                self.cache.move_to_end(key)
                self.stats['hits'] += 1
                return value

            self.stats['misses'] += 1
            return None

    def set(self, key: str, value: Any, ttl: int = None) -> None:
        """设置缓存项"""
        with self.lock:
            # 计算内存占用
            value_size = len(pickle.dumps(value))

            # 检查是否需要清理空间
            while (len(self.cache) >= self.max_size or
                   self.current_memory + value_size > self.max_memory):
                if not self._evict_lru():
                    break

            # 删除已存在的键
            if key in self.cache:
                self._remove_item(key)

            # 计算过期时间
            expire_time = time.time() + ttl if ttl else None

            # 添加新项
            self.cache[key] = (value, expire_time)
            self.current_memory += value_size
            self.cache.move_to_end(key)

    def _evict_lru(self) -> bool:
        """清理最久未使用的项"""
        if not self.cache:
            return False

        key, (value, expire_time) = self.cache.popitem(last=False)
        self.current_memory -= len(pickle.dumps(value))
        self.stats['evictions'] += 1
        return True

    def _remove_item(self, key: str) -> None:
        """移除指定项"""
        if key in self.cache:
            value, _ = self.cache.pop(key)
            self.current_memory -= len(pickle.dumps(value))

class LocalCacheManager:
    """本地缓存管理器"""

    def __init__(self):
        self.caches = {
            'user_data': LRUCache(max_size=1000, max_memory=50 * 1024 * 1024),
            'strategy_data': LRUCache(max_size=500, max_memory=30 * 1024 * 1024),
            'market_data': LRUCache(max_size=2000, max_memory=100 * 1024 * 1024)
        }

    def get_cache(self, cache_type: str) -> LRUCache:
        """获取指定类型的缓存"""
        return self.caches.get(cache_type, self.caches['user_data'])
```

### 3. 多层级缓存策略

```python
import asyncio
import hashlib
from typing import Any, Callable
from functools import wraps

class MultiLevelCache:
    """多级缓存系统"""

    def __init__(self, redis_manager: RedisCacheManager, local_manager: LocalCacheManager):
        self.redis = redis_manager
        self.local = local_manager
        self.cache_stats = {
            'l1_hits': 0,
            'l2_hits': 0,
            'l3_hits': 0,
            'total_requests': 0
        }

    def get(self, key: str, cache_type: str = 'user_data') -> Optional[Any]:
        """多级缓存获取数据"""
        self.cache_stats['total_requests'] += 1

        # L1缓存（本地缓存）
        local_cache = self.local.get_cache(cache_type)
        data = local_cache.get(key)
        if data is not None:
            self.cache_stats['l1_hits'] += 1
            return data

        # L2缓存（Redis缓存）
        data = self.redis.get(key)
        if data is not None:
            self.cache_stats['l2_hits'] += 1
            # 回填到L1缓存
            local_cache.set(key, data, ttl=300)  # 5分钟TTL
            return data

        # L3缓存（数据库）
        self.cache_stats['l3_hits'] += 1
        return None

    def set(self, key: str, value: Any, cache_type: str = 'user_data',
            l1_ttl: int = 300, l2_ttl: int = 3600) -> None:
        """多级缓存设置数据"""
        # 设置L1缓存
        local_cache = self.local.get_cache(cache_type)
        local_cache.set(key, value, ttl=l1_ttl)

        # 设置L2缓存
        self.redis.set(key, value, ttl=l2_ttl)

    def invalidate(self, key: str) -> None:
        """缓存失效"""
        # 清除L1缓存
        for cache in self.local.caches.values():
            if key in cache.cache:
                cache._remove_item(key)

        # 清除L2缓存
        self.redis.delete(key)

def cache_result(key_prefix: str, cache_type: str = 'user_data',
                l1_ttl: int = 300, l2_ttl: int = 3600):
    """缓存装饰器"""
    def decorator(func: Callable):
        @wraps(func)
        async def wrapper(*args, **kwargs):
            # 生成缓存键
            key_data = f"{key_prefix}:{args}:{kwargs}"
            cache_key = hashlib.md5(key_data.encode()).hexdigest()

            # 尝试从缓存获取
            cache_manager = get_cache_manager()
            result = cache_manager.get(cache_key, cache_type)

            if result is not None:
                return result

            # 执行原函数
            if asyncio.iscoroutinefunction(func):
                result = await func(*args, **kwargs)
            else:
                result = func(*args, **kwargs)

            # 设置缓存
            cache_manager.set(cache_key, result, cache_type, l1_ttl, l2_ttl)

            return result

        return wrapper
    return decorator
```

### 4. 缓存保护机制

```python
import asyncio
import time
from typing import Any, Optional

class CacheProtection:
    """缓存保护机制"""

    def __init__(self, cache_manager: MultiLevelCache):
        self.cache = cache_manager
        self.bloom_filter = set()  # 简化版布隆过滤器
        self.hot_data_keys = set()
        self.lock_cache = {}

    async def get_with_protection(self, key: str, cache_type: str = 'user_data',
                                fallback_func: Callable = None) -> Optional[Any]:
        """带保护机制的缓存获取"""

        # 1. 空值缓存（防止缓存穿透）
        null_key = f"null:{key}"
        if self.cache.redis.get(null_key):
            return None

        # 2. 检查是否为热点数据（防止缓存击穿）
        if key in self.hot_data_keys:
            return await self._get_hot_data_with_lock(key, cache_type, fallback_func)

        # 3. 正常缓存获取
        result = self.cache.get(key, cache_type)
        if result is not None:
            # 更新热点数据标记
            await self._update_hot_data_mark(key)
            return result

        # 4. 缓存未命中时的保护处理
        if fallback_func:
            # 使用分布式锁防止缓存击穿
            result = await self._get_with_lock(key, cache_type, fallback_func)
            return result

        # 5. 设置空值缓存
        self.cache.redis.setex(null_key, 60, "null")
        return None

    async def _get_hot_data_with_lock(self, key: str, cache_type: str,
                                    fallback_func: Callable) -> Any:
        """热点数据加锁获取"""
        lock_key = f"lock:{key}"

        # 尝试获取分布式锁
        if await self._acquire_lock(lock_key, timeout=10):
            try:
                # 双重检查
                result = self.cache.get(key, cache_type)
                if result is not None:
                    return result

                # 执行回退函数
                result = await fallback_func()

                # 设置缓存，热点数据永不过期
                self.cache.set(key, result, cache_type, l1_ttl=0, l2_ttl=86400)

                return result
            finally:
                await self._release_lock(lock_key)
        else:
            # 等待并重试
            await asyncio.sleep(0.1)
            return self.cache.get(key, cache_type) or await fallback_func()

    async def _update_hot_data_mark(self, key: str) -> None:
        """更新热点数据标记"""
        access_count_key = f"access:{key}"
        current_count = self.cache.redis.get(access_count_key) or 0

        if int(current_count) >= 10:  # 10次访问后标记为热点
            self.hot_data_keys.add(key)
            self.cache.redis.setex(access_count_key, 3600, "hot")
        else:
            self.cache.redis.incr(access_count_key)
            self.cache.redis.expire(access_count_key, 300)  # 5分钟过期

    async def _acquire_lock(self, lock_key: str, timeout: int = 10) -> bool:
        """获取分布式锁"""
        end_time = time.time() + timeout
        while time.time() < end_time:
            if self.cache.redis.setnx(lock_key, "locked"):
                self.cache.redis.expire(lock_key, timeout)
                return True
            await asyncio.sleep(0.01)
        return False

    async def _release_lock(self, lock_key: str) -> None:
        """释放分布式锁"""
        self.cache.redis.delete(lock_key)

# 缓存监控和管理
class CacheMonitor:
    """缓存监控器"""

    def __init__(self, cache_manager: MultiLevelCache):
        self.cache = cache_manager
        self.metrics = {
            'hit_rate': 0,
            'avg_response_time': 0,
            'memory_usage': 0,
            'eviction_rate': 0
        }

    def get_cache_stats(self) -> dict:
        """获取缓存统计信息"""
        total_requests = self.cache.cache_stats['total_requests']
        if total_requests == 0:
            return self.metrics

        l1_hits = self.cache.cache_stats['l1_hits']
        l2_hits = self.cache.cache_stats['l2_hits']

        # 计算命中率
        total_hits = l1_hits + l2_hits
        hit_rate = (total_hits / total_requests) * 100 if total_requests > 0 else 0

        # 获取内存使用情况
        memory_usage = sum(
            cache.current_memory for cache in self.cache.local.caches.values()
        )

        return {
            'hit_rate': round(hit_rate, 2),
            'l1_hit_rate': round((l1_hits / total_requests) * 100, 2),
            'l2_hit_rate': round((l2_hits / total_requests) * 100, 2),
            'l3_hit_rate': round((self.cache.cache_stats['l3_hits'] / total_requests) * 100, 2),
            'memory_usage': memory_usage,
            'memory_usage_mb': round(memory_usage / (1024 * 1024), 2),
            'total_requests': total_requests,
            'cache_size': {
                name: len(cache.cache)
                for name, cache in self.cache.local.caches.items()
            }
        }

    def clear_cache(self, cache_type: str = None) -> dict:
        """清理缓存"""
        result = {'cleared': [], 'errors': []}

        try:
            if cache_type:
                # 清理指定类型缓存
                if cache_type in self.cache.local.caches:
                    self.cache.local.caches[cache_type].cache.clear()
                    self.cache.local.caches[cache_type].current_memory = 0
                    result['cleared'].append(f"local:{cache_type}")
            else:
                # 清理所有本地缓存
                for name, cache in self.cache.local.caches.items():
                    cache.cache.clear()
                    cache.current_memory = 0
                    result['cleared'].append(f"local:{name}")

            # 重置统计信息
            for cache in self.cache.local.caches.values():
                cache.stats = {'hits': 0, 'misses': 0, 'evictions': 0}

        except Exception as e:
            result['errors'].append(str(e))

        return result
```
