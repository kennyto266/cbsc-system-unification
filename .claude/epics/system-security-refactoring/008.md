## Description

为CBSC量化交易系统构建全方位的监控和安全审计系统，包括Prometheus+Grafana性能监控、ELK日志聚合、安全审计追踪和智能告警系统。通过实时监控、自动化告警和全面的审计日志，确保系统的稳定运行、安全合规和快速故障定位能力。

```python
# 应用程序指标收集器
from prometheus_client import Counter, Histogram, Gauge, CollectorRegistry, generate_latest
from prometheus_client.exposition import start_http_server
import time
import psutil
from typing import Dict, Any
from functools import wraps

class PrometheusMetricsCollector:
    """Prometheus指标收集器"""

    def __init__(self, app_name: str = "cbsc-trading"):
        self.app_name = app_name
        self.registry = CollectorRegistry()

        # HTTP请求指标
        self.http_requests_total = Counter(
            'http_requests_total',
            'Total HTTP requests',
            ['method', 'endpoint', 'status_code'],
            registry=self.registry
        )

        self.http_request_duration = Histogram(
            'http_request_duration_seconds',
            'HTTP request duration',
            ['method', 'endpoint'],
            registry=self.registry,
            buckets=[0.1, 0.25, 0.5, 1.0, 2.5, 5.0, 10.0]
        )

        # 业务指标
        self.strategy_executions_total = Counter(
            'strategy_executions_total',
            'Total strategy executions',
            ['strategy_type', 'status'],
            registry=self.registry
        )

        self.trading_volume = Counter(
            'trading_volume_total',
            'Total trading volume',
            ['symbol', 'side'],
            registry=self.registry
        )

        self.active_users = Gauge(
            'active_users_total',
            'Number of active users',
            registry=self.registry
        )

        self.system_memory_usage = Gauge(
            'system_memory_usage_bytes',
            'System memory usage in bytes',
            registry=self.registry
        )

        self.system_cpu_usage = Gauge(
            'system_cpu_usage_percent',
            'System CPU usage percentage',
            registry=self.registry
        )

        # 数据库连接指标
        self.db_connections_active = Gauge(
            'db_connections_active',
            'Active database connections',
            registry=self.registry
        )

        self.db_query_duration = Histogram(
            'db_query_duration_seconds',
            'Database query duration',
            ['query_type'],
            registry=self.registry
        )

    def record_http_request(self, method: str, endpoint: str, status_code: int, duration: float):
        """记录HTTP请求指标"""
        self.http_requests_total.labels(
            method=method,
            endpoint=endpoint,
            status_code=str(status_code)
        ).inc()

        self.http_request_duration.labels(
            method=method,
            endpoint=endpoint
        ).observe(duration)

    def record_strategy_execution(self, strategy_type: str, status: str):
        """记录策略执行指标"""
        self.strategy_executions_total.labels(
            strategy_type=strategy_type,
            status=status
        ).inc()

    def record_trading_volume(self, symbol: str, side: str, volume: float):
        """记录交易量指标"""
        self.trading_volume.labels(
            symbol=symbol,
            side=side
        ).inc(volume)

    def update_system_metrics(self):
        """更新系统指标"""
        # 内存使用
        memory = psutil.virtual_memory()
        self.system_memory_usage.set(memory.used)

        # CPU使用
        cpu_percent = psutil.cpu_percent(interval=1)
        self.system_cpu_usage.set(cpu_percent)

    def get_metrics(self) -> str:
        """获取Prometheus格式的指标"""
        self.update_system_metrics()
        return generate_latest(self.registry).decode('utf-8')

    def start_metrics_server(self, port: int = 8000):
        """启动指标服务器"""
        start_http_server(port, registry=self.registry)

# 指标收集装饰器
def track_metrics(metrics_collector: PrometheusMetricsCollector):
    """指标收集装饰器"""
    def decorator(func):
        @wraps(func)
        async def wrapper(*args, **kwargs):
            start_time = time.time()
            method = getattr(args[0], 'method', 'unknown') if args else 'unknown'
            endpoint = getattr(args[0], 'path', 'unknown') if args else 'unknown'

            try:
                result = await func(*args, **kwargs)
                status_code = getattr(result, 'status_code', 200)
                return result
            except Exception as e:
                status_code = 500
                raise
            finally:
                duration = time.time() - start_time
                metrics_collector.record_http_request(method, endpoint, status_code, duration)

        return wrapper
    return decorator
```

```python
class SecurityAuditManager:
    """安全审计管理器"""

    def __init__(self, logger: StructuredLogger):
        self.logger = logger
        self.suspicious_patterns = []
        self._setup_suspicious_patterns()

    def _setup_suspicious_patterns(self):
        """设置可疑行为模式"""
        self.suspicious_patterns = [
            {

                "condition": lambda logs: self._check_frequent_failed_logins(logs),
                "risk_score": 0.8
            },
            {
                "name": "异常登录地点",
                "condition": lambda logs: self._check_abnormal_location(logs),
                "risk_score": 0.7
            },
            {
                "name": "敏感数据访问",
                "condition": lambda logs: self._check_sensitive_data_access(logs),
                "risk_score": 0.6
            }
        ]

    def audit_user_action(self, user_id: str, action: str, resource: str,
                         result: str, user_ip: str, session_id: str,
                         details: Dict[str, Any] = None):
        """审计用户操作"""
        audit_log = SecurityAuditLog(
            timestamp=time.time(),
            event_type=SecurityEventType.DATA_ACCESS,
            user_id=user_id,
            user_ip=user_ip,
            action=action,
            resource=resource,
            result=result,
            details=details or {},
            session_id=session_id,
            risk_score=self._calculate_risk_score(user_id, action, resource)
        )

        self.logger.log_security_event(audit_log)

    def audit_login_attempt(self, user_id: str, user_ip: str, success: bool,
                           user_agent: str = None, session_id: str = None):
        """审计登录尝试"""
        audit_log = SecurityAuditLog(
            timestamp=time.time(),
            event_type=SecurityEventType.LOGIN_ATTEMPT,
            user_id=user_id,
            user_ip=user_ip,
            action="login",
            resource="auth_service",
            result="success" if success else "failed",
            details={
                "user_agent": user_agent,
                "success": success
            },
            session_id=session_id or "",
            risk_score=0.0 if success else 0.5
        )

        self.logger.log_security_event(audit_log)

    def detect_suspicious_activity(self, user_logs: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
        """检测可疑活动"""
        suspicious_activities = []

        for pattern in self.suspicious_patterns:
            if pattern["condition"](user_logs):
                suspicious_activities.append({
                    "pattern_name": pattern["name"],
                    "risk_score": pattern["risk_score"],
                    "detected_at": time.time()
                })

        """检查频繁登录失败"""
        recent_failures = [
            log for log in logs
            if log.get("action") == "login" and log.get("result") == "failed"
            and time.time() - log.get("timestamp", 0) < 3600  # 1小时内
        ]
        return len(recent_failures) >= 5

    def _check_abnormal_location(self, logs: List[Dict[str, Any]]) -> bool:
        """检查异常登录地点"""
        # 简化的异常地点检测
        recent_ips = set(
            log.get("user_ip") for log in logs
            if time.time() - log.get("timestamp", 0) < 86400  # 24小时内
        )
        return len(recent_ips) > 3

    def _check_sensitive_data_access(self, logs: List[Dict[str, Any]]) -> bool:
        """检查敏感数据访问"""
        sensitive_resources = ["user_data", "trading_config", "api_keys"]

            log.get("resource") in sensitive_resources
            for log in logs
```

```python
import asyncio
import smtplib
from email.mime.text import MIMEText
from email.mime.multipart import MIMEMultipart
from typing import Dict, Any, List, Callable
from dataclasses import dataclass
from enum import Enum
import requests

class AlertLevel(Enum):
    INFO = "info"
    WARNING = "warning"
    CRITICAL = "critical"
    EMERGENCY = "emergency"

class AlertChannel(Enum):
    EMAIL = "email"
    SMS = "sms"
    WEBHOOK = "webhook"
    DINGTALK = "dingtalk"
    WECHAT = "wechat"

@dataclass
class AlertRule:
    """告警规则"""
    name: str
    condition: str  # 告警条件表达式
    level: AlertLevel
    duration: int  # 持续时间（秒）
    channels: List[AlertChannel]
    enabled: bool = True
    cooldown: int = 300  # 冷却时间（秒）

@dataclass
class Alert:
    """告警事件"""
    rule_name: str
    level: AlertLevel
    message: str
    timestamp: float
    details: Dict[str, Any]
    status: str = "active"

class AlertManager:
    """告警管理器"""

    def __init__(self, config: Dict[str, Any]):
        self.config = config
        self.rules: Dict[str, AlertRule] = {}
        self.active_alerts: Dict[str, Alert] = {}
        self.alert_history: List[Alert] = []
        self.last_notifications: Dict[str, float] = {}

        # 告警处理器
        self.handlers = {
            AlertChannel.EMAIL: self._send_email_alert,
            AlertChannel.SMS: self._send_sms_alert,
            AlertChannel.WEBHOOK: self._send_webhook_alert,
            AlertChannel.DINGTALK: self._send_dingtalk_alert,
            AlertChannel.WECHAT: self._send_wechat_alert
        }

    def add_rule(self, rule: AlertRule):
        """添加告警规则"""
        self.rules[rule.name] = rule

    async def check_metrics(self, metrics: Dict[str, float]):
        """检查指标并触发告警"""
        for rule_name, rule in self.rules.items():
            if not rule.enabled:
                continue

            try:
                # 评估告警条件
                if self._evaluate_condition(rule.condition, metrics):
                    await self._trigger_alert(rule_name, rule, metrics)
                else:
                    await self._resolve_alert(rule_name)

            except Exception as e:
                print(f"Error checking rule {rule_name}: {str(e)}")

    def _evaluate_condition(self, condition: str, metrics: Dict[str, float]) -> bool:
        """评估告警条件"""
        try:
            # 替换指标变量
            for key, value in metrics.items():
                condition = condition.replace(f"${key}", str(value))

            # 安全评估表达式
            allowed_names = {
                "__builtins__": {},
                "abs": abs,
                "max": max,
                "min": min,
                "round": round
            }

            return eval(condition, allowed_names, {})
        except Exception:
            return False

    async def _trigger_alert(self, rule_name: str, rule: AlertRule, metrics: Dict[str, Any]):
        """触发告警"""
        current_time = time.time()

        # 检查冷却时间
        if rule_name in self.last_notifications:
            if current_time - self.last_notifications[rule_name] < rule.cooldown:
                return

        # 检查是否已有活跃告警
        if rule_name in self.active_alerts:
            alert = self.active_alerts[rule_name]
            # 更新持续时间
            if current_time - alert.timestamp >= rule.duration:
                await self._send_notifications(alert, rule)
                self.last_notifications[rule_name] = current_time
        else:
            # 创建新告警
            alert = Alert(
                rule_name=rule_name,
                level=rule.level,
                message=f"告警规则 {rule_name} 被触发",
                timestamp=current_time,
                details=metrics
            )
            self.active_alerts[rule_name] = alert

            # 立即发送紧急告警
            if rule.level in [AlertLevel.CRITICAL, AlertLevel.EMERGENCY]:
                await self._send_notifications(alert, rule)
                self.last_notifications[rule_name] = current_time

    async def _resolve_alert(self, rule_name: str):
        """解决告警"""
        if rule_name in self.active_alerts:
            alert = self.active_alerts[rule_name]
            alert.status = "resolved"
            self.alert_history.append(alert)
            del self.active_alerts[rule_name]

    async def _send_notifications(self, alert: Alert, rule: AlertRule):
        """发送告警通知"""
        for channel in rule.channels:
            try:
                handler = self.handlers.get(channel)
                if handler:
                    await handler(alert)
            except Exception as e:
                print(f"Failed to send {channel.value} alert: {str(e)}")

    async def _send_email_alert(self, alert: Alert):
        """发送邮件告警"""
        smtp_config = self.config.get("smtp", {})
        recipients = self.config.get("alert_recipients", {}).get("email", [])

        if not recipients:
            return

        subject = f"[{alert.level.value.upper()}] {alert.message}"
        body = self._format_alert_message(alert)

        msg = MIMEMultipart()
        msg['From'] = smtp_config.get("from_email")
        msg['To'] = ", ".join(recipients)
        msg['Subject'] = subject
        msg.attach(MIMEText(body, 'html', 'utf-8'))

        try:
            server = smtplib.SMTP(smtp_config.get("host"), smtp_config.get("port"))
            server.starttls()
            server.login(smtp_config.get("username"), smtp_config.get("password"))
            server.send_message(msg)
            server.quit()
        except Exception as e:
            print(f"Failed to send email alert: {str(e)}")

    async def _send_dingtalk_alert(self, alert: Alert):
        """发送钉钉告警"""
        webhook_url = self.config.get("dingtalk", {}).get("webhook_url")

            response.raise_for_status()
        except Exception as e:
            print(f"Failed to send DingTalk alert: {str(e)}")

    def get_alert_statistics(self) -> Dict[str, Any]:
        """获取告警统计信息"""
        total_alerts = len(self.alert_history)
        active_alerts = len(self.active_alerts)

        level_counts = {}
        for alert in self.alert_history:
            level_counts[alert.level.value] = level_counts.get(alert.level.value, 0) + 1

        return {
            "total_alerts": total_alerts,
            "active_alerts": active_alerts,
            "alert_levels": level_counts,
            "active_rules": len([r for r in self.rules.values() if r.enabled])
        }
```
