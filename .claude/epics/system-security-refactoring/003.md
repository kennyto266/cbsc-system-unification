### 1. 统一输入验证框架
```python
from pydantic import BaseModel, validator, Field
from typing import Any, Dict, List, Optional, Union
import re
import html
import bleach
from fastapi import HTTPException, Request
from slowapi import Limiter, _rate_limit_exceeded_handler

class InputValidator:
    """统一输入验证器"""

    def __init__(self):
        # CBSC特定的验证规则
        self.cbsc_patterns = {
            'strategy_name': r'^[a-zA-Z0-9_-]{1,50}$',
            'symbol': r'^[A-Z0-9]{1,10}$',
            'parameter_name': r'^[a-z_][a-z0-9_]*$',
            'numeric_range': r'^-?\d+(\.\d+)?$',
            'percentage': r'^(100(\.0+)?|\d{1,2}(\.\d+)?)%?$'
        }

        # 危险模式检测
        self.dangerous_patterns = [
            r'<script[^>]*>.*?</script>',  # XSS
            r'javascript:',                # JavaScript协议
            r'on\w+\s*=',                 # 事件处理器
            r'eval\s*\(',                 # eval函数
            r'expression\s*\(',           # CSS表达式
            r'@import',                   # CSS导入
            r'union\s+select',            # SQL注入
            r'drop\s+table',              # SQL删除
            r'exec\s*\(',                 # 代码执行
        ]

    def validate_strategy_input(self, data: Dict[str, Any]) -> Dict[str, Any]:
        """验证策略输入数据"""
        validated_data = {}
        errors = []

        for key, value in data.items():
            try:
                if key == 'strategy_name':
                    validated_data[key] = self._validate_strategy_name(value)
                elif key == 'parameters':
                    validated_data[key] = self._validate_parameters(value)
                elif key == 'symbol':
                    validated_data[key] = self._validate_symbol(value)
                elif key == 'risk_level':
                    validated_data[key] = self._validate_risk_level(value)
                else:
                    validated_data[key] = self._sanitize_input(value)

            except ValidationError as e:
                errors.append(f"{key}: {str(e)}")

        if errors:
            raise HTTPException(
                status_code=422,
                detail={"validation_errors": errors}
            )

        return validated_data

    def _validate_strategy_name(self, name: str) -> str:
        """验证策略名称"""
        if not isinstance(name, str):
            raise ValidationError("策略名称必须是字符串")

        name = name.strip()
        if not name:
            raise ValidationError("策略名称不能为空")

        if len(name) > 50:
            raise ValidationError("策略名称不能超过50个字符")

        if not re.match(self.cbsc_patterns['strategy_name'], name):
            raise ValidationError("策略名称只包含字母、数字、下划线和连字符")

        # XSS防护
        name = html.escape(name)
        return name

    def _validate_parameters(self, params: Any) -> Dict[str, Any]:
        """验证策略参数"""
        if not isinstance(params, dict):
            raise ValidationError("参数必须是字典格式")

        validated_params = {}
        for param_name, param_value in params.items():
            # 验证参数名
            if not re.match(self.cbsc_patterns['parameter_name'], param_name):
                raise ValidationError(f"参数名 '{param_name}' 格式无效")

            # 验证参数值
            validated_params[param_name] = self._validate_parameter_value(param_value)

        return validated_params

    def _validate_parameter_value(self, value: Any) -> Union[float, int, str, bool]:
        """验证参数值"""
        if isinstance(value, (int, float)):
            # 数值验证
            if abs(value) > 1e10:  # 防止数值溢出
                raise ValidationError("参数值超出允许范围")
            return value
        elif isinstance(value, str):
            # 字符串验证
            if len(value) > 1000:
                raise ValidationError("字符串参数值过长")
            return self._sanitize_input(value)
        elif isinstance(value, bool):
            return value
        else:
            raise ValidationError("参数值类型不支持")

    def _sanitize_input(self, value: str) -> str:
        """清理输入数据"""
        if not isinstance(value, str):
            return value

        # 检查危险模式
        for pattern in self.dangerous_patterns:
            if re.search(pattern, value, re.IGNORECASE | re.DOTALL):
                raise ValidationError("输入包含不安全的内容")

        # HTML清理
        cleaned_value = bleach.clean(
            value,
            tags=[],  # 不允许任何HTML标签
            attributes={},
            strip=True
        )

        return cleaned_value.strip()

class SecurityHeadersMiddleware:
```

### 2. CSRF防护机制
```python
import secrets
from fastapi import Cookie, Header, HTTPException, Depends
from fastapi.security import HTTPBearer, HTTPAuthorizationCredentials

class CSRFProtection:
    """CSRF防护"""

    def __init__(self):
        self.token_length = 32
        self.cookie_name = "cbsc-csrf-token"
        self.header_name = "X-CSRF-Token"

    def generate_csrf_token(self) -> str:
        """生成CSRF令牌"""
        return secrets.token_urlsafe(self.token_length)

    def validate_csrf_token(self,
                          request: Request,
                          csrf_token: str = Header(None, alias=self.header_name),
                          cookie_token: str = Cookie(None, alias=self.cookie_name)) -> bool:
        """验证CSRF令牌"""
        # 对于安全方法(GET, HEAD, OPTIONS)不需要CSRF验证
        if request.method in ["GET", "HEAD", "OPTIONS"]:
            return True

        if not csrf_token or not cookie_token:
            raise HTTPException(
                status_code=403,
                detail="CSRF token missing"
            )

        if not secrets.compare_digest(csrf_token, cookie_token):
            raise HTTPException(
                status_code=403,
                detail="CSRF token invalid"
            )

        return True

def csrf_protect(request: Request = Depends(),
                 csrf_service: CSRFProtection = Depends()) -> bool:
    """CSRF保护装饰器"""
    return csrf_service.validate_csrf_token(request)
```

### 3. API限流和DDoS防护
```python
from slowapi import Limiter
from slowapi.util import get_remote_address
from slowapi.errors import RateLimitExceeded
import redis
from collections import defaultdict
import time

class AdvancedRateLimiter:
    """高级API限流器"""

    def __init__(self):
        self.redis_client = redis.Redis(host='localhost', port=6379, db=1)

        # 限流配置
        self.rate_limits = {
            'auth': {'requests': 5, 'window': 60},      # 认证接口：5次/分钟
            'trading': {'requests': 100, 'window': 60}, # 交易接口：100次/分钟
            'api': {'requests': 1000, 'window': 60},    # 一般API：1000次/分钟
            'upload': {'requests': 10, 'window': 300},  # 文件上传：10次/5分钟
        }

        self.blacklist_duration = 3600  # 黑名单持续1小时
        self.suspicious_threshold = 100  # 可疑活动阈值

    def is_rate_limited(self, identifier: str, endpoint_type: str) -> bool:
        """检查是否触发限流"""
        config = self.rate_limits.get(endpoint_type, self.rate_limits['api'])

        # 检查黑名单
        if self._is_blacklisted(identifier):
            return True

        # 滑动窗口限流
        current_time = int(time.time())
        window = config['window']
        max_requests = config['requests']

        # 清理过期记录
        self._cleanup_old_requests(identifier, current_time - window)

        # 检查当前窗口请求数
        request_count = self._get_request_count(identifier, current_time - window)

        if request_count >= max_requests:
            # 记录可疑活动
            self._record_suspicious_activity(identifier)

            # 如果超过阈值，加入黑名单
            if request_count > max_requests * 2:
                self._blacklist_identifier(identifier)

            return True

        # 记录当前请求
        self._record_request(identifier, current_time)
        return False

    def _is_blacklisted(self, identifier: str) -> bool:
        """检查是否在黑名单中"""
        blacklist_key = f"blacklist:{identifier}"
        return self.redis_client.exists(blacklist_key)

    def _blacklist_identifier(self, identifier: str):
        """将标识符加入黑名单"""
        blacklist_key = f"blacklist:{identifier}"
        self.redis_client.setex(blacklist_key, self.blacklist_duration, "1")

    def _record_suspicious_activity(self, identifier: str):
        """记录可疑活动"""
        suspicious_key = f"suspicious:{identifier}"
        current_count = self.redis_client.get(suspicious_key) or 0
        self.redis_client.setex(suspicious_key, 3600, int(current_count) + 1)
```

### 4. 文件上传安全检查
```python
import magic
from pathlib import Path
from werkzeug.utils import secure_filename

class SecureFileUpload:
    """安全文件上传处理"""

    def __init__(self):
        # 允许的文件类型
        self.allowed_extensions = {
            'csv': ['text/csv', 'application/csv'],
            'json': ['application/json'],
            'xlsx': ['application/vnd.openxmlformats-officedocument.spreadsheetml.sheet'],
            'pdf': ['application/pdf'],
            'txt': ['text/plain'],
        }

        # 文件大小限制 (MB)
        self.max_file_sizes = {
            'csv': 50,
            'json': 10,
            'xlsx': 100,
            'pdf': 20,
            'txt': 10,
        }

        self.upload_dir = Path("uploads/secure")
        self.upload_dir.mkdir(parents=True, exist_ok=True)

    def validate_file(self, file_content: bytes, filename: str) -> Dict[str, Any]:
        """验证上传文件"""
        result = {
            'valid': False,
            'errors': [],
            'safe_filename': None
        }

        try:
            # 1. 文件名安全处理
            safe_filename = secure_filename(filename)
            if not safe_filename:
                result['errors'].append("文件名包含不安全字符")
                return result

            # 2. 文件扩展名检查
            file_ext = Path(safe_filename).suffix.lower().lstrip('.')
            if file_ext not in self.allowed_extensions:
                result['errors'].append(f"不支持的文件类型: {file_ext}")
                return result

            # 3. 文件大小检查
            file_size = len(file_content) / (1024 * 1024)  # MB
            max_size = self.max_file_sizes.get(file_ext, 10)
            if file_size > max_size:
                result['errors'].append(f"文件大小超过限制 ({max_size}MB)")
                return result

            # 4. MIME类型检查
            mime_type = magic.from_buffer(file_content, mime=True)
            allowed_mimes = self.allowed_extensions[file_ext]
            if mime_type not in allowed_mimes:
                result['errors'].append(f"文件类型不匹配: {mime_type}")
                return result

            # 5. 恶意内容扫描
            if self._scan_malicious_content(file_content):
                result['errors'].append("文件包含恶意内容")
                return result

            # 6. 宏病毒检查 (针对Office文档)
            if file_ext in ['xlsx'] and self._has_macros(file_content):
                result['errors'].append("Office文件包含宏，存在安全风险")
                return result

            result['valid'] = True
            result['safe_filename'] = safe_filename
            result['mime_type'] = mime_type
            result['file_size'] = file_size

        except Exception as e:
            result['errors'].append(f"文件验证失败: {str(e)}")

        return result

    def _scan_malicious_content(self, content: bytes) -> bool:
        """扫描恶意内容"""
        # 检查常见恶意模式
        malicious_patterns = [
            b'<script',
            b'javascript:',
            b'vbscript:',
            b'data:text/html',
            b'<?php',
            b'<%',
            b'eval(',
            b'exec(',
            b'system(',
        ]

        content_lower = content.lower()
        for pattern in malicious_patterns:
            if pattern in content_lower:
                return True

        return False

    def _has_macros(self, content: bytes) -> bool:
        """检查Office文件是否包含宏"""
        # 简化的宏检测逻辑
        macro_indicators = [
            b'VBA',
            b'macro',
            b'AutoOpen',
            b'AutoClose',
            b'Document_Open',
        ]

        for indicator in macro_indicators:
            if indicator in content:
                return True

        return False

    async def save_file(self, file_content: bytes, safe_filename: str) -> str:
        """安全保存文件"""
        # 生成唯一文件名
        timestamp = int(time.time())
        unique_id = secrets.token_hex(8)
        final_filename = f"{timestamp}_{unique_id}_{safe_filename}"

        file_path = self.upload_dir / final_filename

        # 安全保存
        with open(file_path, 'wb') as f:
            f.write(file_content)

        # 设置安全权限
        file_path.chmod(0o644)

        return str(file_path)
```

### 5. 数据泄露防护(DLP)
```python
import re
from typing import List, Pattern

class DataLeakagePrevention:
    """数据泄露防护"""

    def __init__(self):
        # 敏感数据模式
        self.sensitive_patterns = {
            'email': re.compile(r'\b[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z|a-z]{2,}\b'),
            'phone': re.compile(r'\b\d{3}-\d{3}-\d{4}\b'),
            'ssn': re.compile(r'\b\d{3}-\d{2}-\d{4}\b'),
            'credit_card': re.compile(r'\b\d{4}[\s-]?\d{4}[\s-]?\d{4}[\s-]?\d{4}\b'),
            'api_key': re.compile(r'\b[A-Za-z0-9]{20,}\b'),
            'password': re.compile(r'(?i)password\s*[:=]\s*[^\s\}]{8,}'),
            'token': re.compile(r'(?i)token\s*[:=]\s*[^\s\}]{20,}'),
        }

        # CBSC特定敏感信息
        self.cbsc_sensitive_patterns = {
            'strategy_secret': re.compile(r'(?i)strategy.*secret\s*[:=]\s*[^\s\}]{10,}'),
            'trading_account': re.compile(r'(?i)account.*number\s*[:=]\s*\d{8,}'),
            'private_key': re.compile(r'-----BEGIN\s+(RSA\s+)?PRIVATE\s+KEY-----'),
        }

    def scan_for_sensitive_data(self, content: str) -> Dict[str, List[str]]:
        """扫描内容中的敏感数据"""
        findings = {}

        # 扫描通用敏感模式
        for pattern_name, pattern in self.sensitive_patterns.items():
            matches = pattern.findall(content)
            if matches:
                findings[pattern_name] = matches

        # 扫描CBSC特定模式
        for pattern_name, pattern in self.cbsc_sensitive_patterns.items():
            matches = pattern.findall(content)
            if matches:
                findings[pattern_name] = matches

        return findings

    def mask_sensitive_data(self, content: str) -> str:
        """掩码敏感数据"""
        masked_content = content

        for pattern_name, pattern in self.sensitive_patterns.items():
            if pattern_name in ['email', 'phone']:
                # 部分掩码
                masked_content = pattern.sub(lambda m: self._partial_mask(m.group()), masked_content)
            else:
                # 完全掩码
                masked_content = pattern.sub('[MASKED]', masked_content)

        return masked_content

    def _partial_mask(self, text: str) -> str:
        """部分掩码文本"""
        if len(text) <= 4:
            return '[MASKED]'

        # 保留前2位和后2位
        return text[:2] + '*' * (len(text) - 4) + text[-2:]

    def validate_data_export(self, user_role: str, data_type: str, record_count: int) -> bool:
        """验证数据导出权限"""
        # 导出权限配置
        export_limits = {
            'admin': {'max_records': 100000, 'allowed_types': ['all']},
            'trader': {'max_records': 10000, 'allowed_types': ['performance', 'strategies']},
            'analyst': {'max_records': 5000, 'allowed_types': ['performance', 'strategies']},
            'viewer': {'max_records': 1000, 'allowed_types': ['performance']},
        }

        if user_role not in export_limits:
            return False

        limits = export_limits[user_role]

        # 检查数据类型权限
        if data_type not in limits['allowed_types'] and 'all' not in limits['allowed_types']:
            return False

        # 检查记录数量限制
        if record_count > limits['max_records']:
            return False

        return True
```
