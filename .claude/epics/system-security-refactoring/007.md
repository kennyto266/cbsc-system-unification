```python
    def __init__(self, app, minimum_size: int = 1024):
        self.app = app
        self.minimum_size = minimum_size

    async def __call__(self, scope, receive, send):
        if scope["type"] != "http":
            await self.app(scope, receive, send)
            return

        # 检查客户端是否支持压缩
        headers = dict(scope.get("headers", []))
        accept_encoding = headers.get(b"accept-encoding", b"").decode()

        if "gzip" not in accept_encoding.lower():
            await self.app(scope, receive, send)
            return

        # 包装发送函数
        response_body = b""
        response_headers = {}
        status_code = None

        async def send_wrapper(message):
            nonlocal response_body, response_headers, status_code
            if message["type"] == "http.response.start":
                status_code = message["status"]
                response_headers = dict(message.get("headers", []))
                await send(message)
            elif message["type"] == "http.response.body":
                response_body += message.get("body", b"")
                if not message.get("more_body", False):
                    # 压缩响应体
                    if len(response_body) >= self.minimum_size:
                        import gzip
                        compressed = gzip.compress(response_body, compresslevel=6)

                        # 更新Content-Length和添加Content-Encoding
                        response_headers[b"content-length"] = str(len(compressed)).encode()
                        response_headers.append((b"content-encoding", b"gzip"))

                        # 发送压缩后的数据
                        await send({
                            "type": "http.response.body",
                            "body": compressed,
                            "more_body": False
                        })
                    else:
                        await send(message)
                else:
                    await send(message)

        await self.app(scope, receive, send_wrapper)
```

### 2. WebSocket性能提升

```python
import asyncio
import json
import time
from typing import Dict, List, Set, Any, Callable
from collections import defaultdict, deque
import websockets
from websockets.server import WebSocketServerProtocol

class HighPerformanceWebSocketManager:
    """高性能WebSocket管理器"""

    def __init__(self):
        self.connections: Dict[str, WebSocketServerProtocol] = {}
        self.subscriptions: Dict[str, Set[str]] = defaultdict(set)  # topic -> connection_ids
        self.message_queue = asyncio.Queue(maxsize=10000)
        self.batch_messages = defaultdict(list)  # topic -> messages
        self.batch_timer = None
        self.metrics = {
            'total_connections': 0,
            'active_connections': 0,
            'messages_sent': 0,
            'message_latency': deque(maxlen=1000),
            'error_count': 0
        }

    async def register_connection(self, websocket: WebSocketServerProtocol,
                                 connection_id: str) -> bool:
        """注册WebSocket连接"""
        try:
            self.connections[connection_id] = websocket
            self.metrics['total_connections'] += 1
            self.metrics['active_connections'] += 1

            # 发送连接确认
            await self._send_message(connection_id, {
                'type': 'connection_confirmed',
                'connection_id': connection_id,
                'timestamp': time.time()
            })

            logger.info(f"WebSocket connection established: {connection_id}")
            return True

        except Exception as e:
            logger.error(f"Failed to register WebSocket connection: {str(e)}")
            self.metrics['error_count'] += 1
            return False

    async def unregister_connection(self, connection_id: str):
        """注销WebSocket连接"""
        if connection_id in self.connections:
            del self.connections[connection_id]
            self.metrics['active_connections'] -= 1

            # 清理订阅
            for topic, subscribers in self.subscriptions.items():
                subscribers.discard(connection_id)

            logger.info(f"WebSocket connection closed: {connection_id}")

    async def subscribe_topic(self, connection_id: str, topic: str) -> bool:
        """订阅主题"""
        if connection_id in self.connections:
            self.subscriptions[topic].add(connection_id)

            # 发送订阅确认
            await self._send_message(connection_id, {
                'type': 'subscription_confirmed',
                'topic': topic,
                'timestamp': time.time()
            })
            return True
        return False

    async def broadcast_message(self, topic: str, message: Dict[str, Any],
                              batch: bool = True):
        """广播消息"""
        if not self.subscriptions[topic]:
            return

        if batch:
            # 批处理模式
            self.batch_messages[topic].append({
                'data': message,
                'timestamp': time.time()
            })

            # 启动批处理定时器
            if self.batch_timer is None:
                self.batch_timer = asyncio.create_task(self._batch_sender())
        else:
            # 立即发送
            await self._send_to_topic(topic, {
                'type': 'message',
                'topic': topic,
                'data': message,
                'timestamp': time.time()
            })

    async def _batch_sender(self):
        """批处理消息发送器"""
        try:
            await asyncio.sleep(0.05)  # 50ms批处理窗口

            for topic, messages in self.batch_messages.items():
                if messages:
                    # 合并消息
                    batch_message = {
                        'type': 'batch_message',
                        'topic': topic,
                        'messages': messages,
                        'batch_size': len(messages),
                        'timestamp': time.time()
                    }
                    await self._send_to_topic(topic, batch_message)
                    messages.clear()

        finally:
            self.batch_timer = None

    async def _send_to_topic(self, topic: str, message: Dict[str, Any]):
        """向主题订阅者发送消息"""
        if not self.subscriptions[topic]:
            return

        failed_connections = set()
        message_data = json.dumps(message)

        for connection_id in self.subscriptions[topic]:
            try:
                websocket = self.connections.get(connection_id)
                if websocket and not websocket.closed:
                    await websocket.send(message_data)
                    self.metrics['messages_sent'] += 1
                else:
                    failed_connections.add(connection_id)

            except Exception as e:
                logger.error(f"Failed to send message to {connection_id}: {str(e)}")
                failed_connections.add(connection_id)
                self.metrics['error_count'] += 1

        # 清理失败的连接
        for connection_id in failed_connections:
            await self.unregister_connection(connection_id)

    async def _send_message(self, connection_id: str, message: Dict[str, Any]):
        """发送单个消息"""
        try:
            websocket = self.connections.get(connection_id)
            if websocket and not websocket.closed:
                await websocket.send(json.dumps(message))
                self.metrics['messages_sent'] += 1
                return True
        except Exception as e:
            logger.error(f"Failed to send message to {connection_id}: {str(e)}")
            self.metrics['error_count'] += 1
        return False

    def get_performance_metrics(self) -> Dict[str, Any]:
        """获取性能指标"""
        avg_latency = 0
        if self.metrics['message_latency']:
            avg_latency = sum(self.metrics['message_latency']) / len(self.metrics['message_latency'])

        return {
            'active_connections': self.metrics['active_connections'],
            'total_connections': self.metrics['total_connections'],
            'messages_sent': self.metrics['messages_sent'],
            'average_latency_ms': round(avg_latency * 1000, 2),
            'error_rate': self.metrics['error_count'] / max(1, self.metrics['messages_sent']),
            'active_topics': len([t for t in self.subscriptions.values() if t])
        }

# 连接池管理器
class WebSocketConnectionPool:
    """WebSocket连接池"""

    def __init__(self, max_connections: int = 10000):
        self.max_connections = max_connections
        self.active_connections = 0
        self.connection_queue = asyncio.Queue(maxsize=max_connections)
        self.connection_stats = defaultdict(int)

    async def acquire_connection(self) -> bool:
        """获取连接槽位"""
        if self.active_connections >= self.max_connections:
            return False

        await self.connection_queue.put(None)
        self.active_connections += 1
        return True

    async def release_connection(self):
        """释放连接槽位"""
        try:
            self.connection_queue.get_nowait()
            self.active_connections -= 1
        except asyncio.QueueEmpty:
            pass
```

```python
import gc
import weakref
import psutil
import asyncio
from typing import Dict, Any, List, Optional
from dataclasses import dataclass
from collections import defaultdict
import tracemalloc

@dataclass
class MemoryStats:
    """内存统计信息"""
    total_mb: float
    used_mb: float
    available_mb: float
    percent: float
    process_memory_mb: float
    gc_objects: int

class MemoryOptimizer:
    """内存优化器"""

    def __init__(self):
        self.object_pools: Dict[str, List] = defaultdict(list)
        self.weak_references: Dict[str, List] = defaultdict(list)
        self.memory_stats_history = deque(maxlen=100)
        self.gc_threshold = 1000  # 对象数量阈值
        self.memory_limit_mb = 2048  # 内存限制

        # 启动内存跟踪
        tracemalloc.start()

    async def optimize_memory(self) -> Dict[str, Any]:
        """执行内存优化"""
        initial_stats = self._get_memory_stats()

        # 1. 强制垃圾回收
        gc.collect()

        # 2. 清理对象池
        await self._cleanup_object_pools()

        # 3. 清理弱引用
        await self._cleanup_weak_references()

        # 4. 压缩内存池
        await self._compress_pools()

        final_stats = self._get_memory_stats()

        # 计算优化效果
        memory_freed = initial_stats.used_mb - final_stats.used_mb
        gc_objects_freed = initial_stats.gc_objects - final_stats.gc_objects

        return {
            'memory_freed_mb': round(memory_freed, 2),
            'gc_objects_freed': gc_objects_freed,
            'optimization_time': time.time(),
            'memory_before': initial_stats,
            'memory_after': final_stats
        }

    def _get_memory_stats(self) -> MemoryStats:
        """获取内存统计信息"""
        memory = psutil.virtual_memory()
        process = psutil.Process()

        return MemoryStats(
            total_mb=memory.total / (1024 * 1024),
            used_mb=memory.used / (1024 * 1024),
            available_mb=memory.available / (1024 * 1024),
            percent=memory.percent,
            process_memory_mb=process.memory_info().rss / (1024 * 1024),
            gc_objects=len(gc.get_objects())
        )

    async def _cleanup_object_pools(self):
        """清理对象池"""
        for pool_name, pool in self.object_pools.items():
            # 保留最近使用的对象
            if len(pool) > 100:
                self.object_pools[pool_name] = pool[-100:]

    async def _cleanup_weak_references(self):
        """清理弱引用"""
        for ref_name, refs in self.weak_references.items():
            # 移除已失效的弱引用
            self.weak_references[ref_name] = [
                ref for ref in refs if ref() is not None
            ]

    async def _compress_pools(self):
        """压缩内存池"""
        for pool_name, pool in self.object_pools.items():
            # 清理列表，释放内存
            pool[:] = [obj for obj in pool if obj is not None]

    def get_object(self, pool_name: str, obj_class: type, *args, **kwargs):
        """从对象池获取对象"""
        pool = self.object_pools[pool_name]

        if pool:
            obj = pool.pop()
            # 重置对象状态
            if hasattr(obj, 'reset'):
                obj.reset()
            return obj
        else:
            # 创建新对象
            return obj_class(*args, **kwargs)

    def return_object(self, pool_name: str, obj: Any):
        """归还对象到池中"""
        pool = self.object_pools[pool_name]
        if len(pool) < 1000:  # 限制池大小
            pool.append(obj)

    def create_weak_reference(self, ref_name: str, obj: Any) -> weakref.ref:
        """创建弱引用"""
        ref = weakref.ref(obj, lambda _: self._remove_weak_reference(ref_name, _))
        self.weak_references[ref_name].append(ref)
        return ref

    def _remove_weak_reference(self, ref_name: str, ref):
        """移除失效的弱引用"""
        if ref_name in self.weak_references:
            try:
                self.weak_references[ref_name].remove(ref)
            except ValueError:
                pass

    async def monitor_memory(self):
        """监控内存使用"""
        while True:
            stats = self._get_memory_stats()
            self.memory_stats_history.append(stats)

            # 检查内存使用率
            if stats.percent > 85:  # 超过85%
                logger.warning(f"High memory usage: {stats.percent}%")
                await self.optimize_memory()

            # 检查进程内存
            if stats.process_memory_mb > self.memory_limit_mb:
                logger.warning(f"Process memory limit exceeded: {stats.process_memory_mb}MB")
                await self.optimize_memory()

            # 检查GC对象数量
            if stats.gc_objects > self.gc_threshold:
                logger.warning(f"High GC object count: {stats.gc_objects}")
                gc.collect()

            await asyncio.sleep(60)  # 每分钟检查一次
```

```python
import asyncio
import time
from typing import Dict, Any, List, Callable
from concurrent.futures import ThreadPoolExecutor, ProcessPoolExecutor
import multiprocessing as mp

class ConcurrencyOptimizer:
    """并发优化器"""

    def __init__(self):
        self.thread_pool = ThreadPoolExecutor(
            max_workers=mp.cpu_count() * 4,
            thread_name_prefix="cbsc_worker"
        )
        self.process_pool = ProcessPoolExecutor(
            max_workers=mp.cpu_count()
        )
        self.semaphores = {}
        self.performance_metrics = {
            'cpu_utilization': 0,
            'active_tasks': 0,
            'completed_tasks': 0,
            'task_duration': deque(maxlen=1000)
        }

    async def execute_task(self, func: Callable, *args,
                          task_type: str = 'cpu_bound',
                          timeout: float = None) -> Any:
        """执行任务（自动选择最优执行方式）"""
        start_time = time.time()
        self.performance_metrics['active_tasks'] += 1

        try:
            if task_type == 'cpu_bound':
                # CPU密集型任务使用进程池
                if asyncio.iscoroutinefunction(func):
                    result = await asyncio.wait_for(func(*args), timeout=timeout)
                else:
                    loop = asyncio.get_event_loop()
                    result = await loop.run_in_executor(
                        self.process_pool, func, *args
                    )
                    if timeout:
                        result = await asyncio.wait_for(asyncio.wrap_future(result), timeout)
            else:
                # I/O密集型任务使用线程池
                loop = asyncio.get_event_loop()
                result = await loop.run_in_executor(
                    self.thread_pool, func, *args
                )

            return result

        except asyncio.TimeoutError:
            logger.error(f"Task timeout after {timeout}s: {func.__name__}")
            raise
        except Exception as e:
            logger.error(f"Task execution failed: {func.__name__}, error: {str(e)}")
            raise
        finally:
            duration = time.time() - start_time
            self.performance_metrics['active_tasks'] -= 1
            self.performance_metrics['completed_tasks'] += 1
            self.performance_metrics['task_duration'].append(duration)

    async def batch_execute(self, tasks: List[Dict[str, Any]],
                          max_concurrent: int = 100) -> List[Any]:
        """批量执行任务"""
        semaphore = asyncio.Semaphore(max_concurrent)

        async def execute_with_semaphore(task_info):
            async with semaphore:
                return await self.execute_task(
                    task_info['func'],
                    *task_info.get('args', []),
                    task_type=task_info.get('type', 'cpu_bound'),
                    timeout=task_info.get('timeout')
                )

        # 并发执行所有任务
        results = await asyncio.gather(
            *[execute_with_semaphore(task) for task in tasks],
            return_exceptions=True
        )

        return results

    def create_semaphore(self, name: str, limit: int) -> asyncio.Semaphore:
        """创建命名信号量"""
        self.semaphores[name] = asyncio.Semaphore(limit)
        return self.semaphores[name]

    async def execute_with_semaphore(self, semaphore_name: str,
                                   func: Callable, *args):
        """使用指定信号量执行任务"""
        semaphore = self.semaphores.get(semaphore_name)
        if not semaphore:
            semaphore = self.create_semaphore(semaphore_name, 100)

        async with semaphore:
            return await self.execute_task(func, *args)

    def get_performance_metrics(self) -> Dict[str, Any]:
        """获取性能指标"""
        avg_duration = 0
        if self.performance_metrics['task_duration']:
            avg_duration = sum(self.performance_metrics['task_duration']) / len(self.performance_metrics['task_duration'])

        # 获取CPU使用率
        cpu_percent = psutil.cpu_percent(interval=1)

        return {
            'cpu_utilization': cpu_percent,
            'active_tasks': self.performance_metrics['active_tasks'],
            'completed_tasks': self.performance_metrics['completed_tasks'],
            'average_task_duration': round(avg_duration, 3),
            'thread_pool_size': self.thread_pool._max_workers,
            'process_pool_size': self.process_pool._max_workers,
            'semaphores': {name: sem._value for name, sem in self.semaphores.items()}
        }

class OptimizedDatabasePool:
    """优化的数据库连接池"""

    def __init__(self, config: Dict[str, Any]):
        self.config = config
        self.pool = None
        self.connection_metrics = {
            'active_connections': 0,
            'idle_connections': 0,
            'total_requests': 0,
            'connection_errors': 0
        }

    async def initialize(self):
        """初始化连接池"""
        import asyncpg

        self.pool = await asyncpg.create_pool(
            self.config['database_url'],
            min_size=self.config.get('min_connections', 10),
            max_size=self.config.get('max_connections', 100),
            max_queries=self.config.get('max_queries', 50000),
            max_inactive_connection_lifetime=self.config.get('max_inactive', 300),
            setup=self._setup_connection,
            init=self._init_connection,
            command_timeout=self.config.get('command_timeout', 60)
        )

    async def _setup_connection(self, conn):
        """设置连接"""
        await conn.set_type_codec(
            'json',
            encoder=str,
            decoder=json.loads,
            schema='pg_catalog'
        )

    async def _init_connection(self, conn):
        """初始化连接"""
        # 设置连接参数
        await conn.execute("SET TIME ZONE 'UTC'")
        await conn.execute("SET statement_timeout = '30s'")

    async def execute_query(self, query: str, *args) -> List[Dict[str, Any]]:
        """执行查询"""
        self.connection_metrics['total_requests'] += 1

        try:
            async with self.pool.acquire() as conn:
                self.connection_metrics['active_connections'] += 1

                # 执行查询
                result = await conn.fetch(query, *args)

                self.connection_metrics['active_connections'] -= 1
                return [dict(row) for row in result]

        except Exception as e:
            self.connection_metrics['connection_errors'] += 1
            self.connection_metrics['active_connections'] -= 1
            raise

    async def get_connection_metrics(self) -> Dict[str, Any]:
        """获取连接指标"""
        if self.pool:
            return {
                'pool_size': self.pool.get_size(),
                'min_size': self.pool.get_min_size(),
                'max_size': self.pool.get_max_size(),
                'active_connections': self.connection_metrics['active_connections'],
                'total_requests': self.connection_metrics['total_requests'],
                'connection_errors': self.connection_metrics['connection_errors'],
                'error_rate': self.connection_metrics['connection_errors'] / max(1, self.connection_metrics['total_requests'])
            }
        return {}
```
