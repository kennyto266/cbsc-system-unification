class MFAService:
    """多因子认证服务"""

    def __init__(self):
        self.otp_secret_base = 'CBSC2025QUANTTRADING'
        self.backup_codes = BackupCodeManager()

    def generate_mfa_secret(self, user_id: int) -> str:
        """为用户生成MFA密钥"""
        secret = pyotp.random_base32()

        # 保存到数据库
        self._save_mfa_secret(user_id, secret)

        return secret

    def generate_qr_code(self, user: User, secret: str) -> bytes:
        """生成MFA QR码"""
        totp_uri = pyotp.totp.TOTP(secret).provisioning_uri(
            name=user.email,
            issuer_name="CBSC量化交易系统"
        )

        qr = qrcode.QRCode(version=1, box_size=10, border=5)
        qr.add_data(totp_uri)
        qr.make(fit=True)

        img = qr.make_image(fill_color="black", back_color="white")
        buffer = BytesIO()
        img.save(buffer, format='PNG')

        return buffer.getvalue()

    def verify_mfa_token(self, user_id: int, token: str) -> bool:
        """验证MFA令牌"""
        secret = self._get_mfa_secret(user_id)
        if not secret:
            return False

        totp = pyotp.TOTP(secret)
        return totp.verify(token, valid_window=1)  # 允许时间窗口

    def generate_backup_codes(self, user_id: int) -> List[str]:
        """生成备用恢复码"""
        codes = [f"{random.randint(100000, 999999)}" for _ in range(10)]
        self._save_backup_codes(user_id, codes)
        return codes

### 3. API权限装饰器
```python
from functools import wraps
from typing import List, Union

def require_permissions(permissions: Union[Permission, List[Permission]]):
    """权限要求装饰器"""
    def decorator(func):
        @wraps(func)
        async def wrapper(*args, **kwargs):
            # 从请求中获取用户信息
            current_user = kwargs.get('current_user')
            if not current_user:
                raise HTTPException(
                    status_code=status.HTTP_401_UNAUTHORIZED,
                    detail="Authentication required"
                )

            # 权限检查
            user_permissions = current_user.get('permissions', [])
            if isinstance(permissions, Permission):
                required_permissions = [permissions]
            else:
                required_permissions = permissions

            # 检查是否具备所需权限
            if not any(perm.value in user_permissions for perm in required_permissions):
                raise HTTPException(
                    status_code=status.HTTP_403_FORBIDDEN,
                    detail=f"Insufficient permissions. Required: {[p.value for p in required_permissions]}"
                )

            return await func(*args, **kwargs)
        return wrapper
    return decorator

def require_role(role: UserRole):
    """角色要求装饰器"""
    def decorator(func):
        @wraps(func)
        async def wrapper(*args, **kwargs):
            current_user = kwargs.get('current_user')
            if not current_user:
                raise HTTPException(
                    status_code=status.HTTP_401_UNAUTHORIZED,
                    detail="Authentication required"
                )

            user_role = current_user.get('role')
            if user_role != role.value:
                raise HTTPException(
                    status_code=status.HTTP_403_FORBIDDEN,
                    detail=f"Role {role.value} required"
                )

            return await func(*args, **kwargs)
        return wrapper
    return decorator
```

class DeviceTrustManager:
    """设备信任管理"""

    def __init__(self):
        self.trusted_devices = RedisDeviceStore()
        self.max_trusted_devices = 5

    def register_device(self, user_id: int, device_info: Dict[str, str]) -> str:
        """注册新设备"""
        device_id = str(uuid.uuid4())
        device_data = {
            'device_id': device_id,
            'user_id': user_id,
            'device_info': device_info,
            'trusted_at': datetime.utcnow(),
            'last_seen': datetime.utcnow(),
            'is_trusted': False
        }

        self.trusted_devices.save_device(device_id, device_data)
        return device_id

    def trust_device(self, user_id: int, device_id: str, mfa_token: str) -> bool:
        """信任设备（需要MFA验证）"""
        if not self.mfa_service.verify_mfa_token(user_id, mfa_token):
            return False

        device = self.trusted_devices.get_device(device_id)
        if device and device['user_id'] == user_id:
            device['is_trusted'] = True
            device['trusted_at'] = datetime.utcnow()
            self.trusted_devices.save_device(device_id, device)
            return True

        return False

    def is_device_trusted(self, user_id: int, device_id: str) -> bool:
        """检查设备是否受信任"""
        device = self.trusted_devices.get_device(device_id)
        return (device and
                device['user_id'] == user_id and
                device['is_trusted'] and
                (datetime.utcnow() - device['last_seen']).days < 30)

class UserSessionManager:
    """用户会话管理"""

    def __init__(self):
        self.active_sessions = RedisSessionStore()
        self.max_sessions_per_user = 3

    def create_session(self, user_id: int, device_id: str, token: str) -> str:
        """创建用户会话"""
        session_id = str(uuid.uuid4())
        session_data = {
            'session_id': session_id,
            'user_id': user_id,
            'device_id': device_id,
            'token_jti': jwt.decode(token, options={"verify_signature": False})['jti'],
            'created_at': datetime.utcnow(),
            'last_activity': datetime.utcnow(),
            'ip_address': self._get_client_ip(),
            'user_agent': self._get_user_agent()
        }

        self.active_sessions.save_session(session_id, session_data)
        return session_id

    def invalidate_user_sessions(self, user_id: int, exclude_session: str = None):
        """使用户所有会话失效"""
        sessions = self.active_sessions.get_user_sessions(user_id)
        for session_id, session_data in sessions.items():
            if session_id != exclude_session:
                # 将token加入黑名单
                self.token_blacklist.blacklist_token(session_data['token_jti'])
                # 删除会话
                self.active_sessions.delete_session(session_id)
