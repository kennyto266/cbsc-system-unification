```python
class SharedMemoryManager:
    def __init__(self):
        self.shared_blocks = {}
        self.access_locks = {}

    def create_shared_array(self, data: np.ndarray, name: str):
        # Create shared memory block accessible by all processes
        shm = shared_memory.SharedMemory(create=True, size=data.nbytes)
        np_array = np.ndarray(data.shape, dtype=data.type, buffer=shm.buf)
        np_array[:] = data[:]
        self.shared_blocks[name] = shm
        return np_array
```

```python
class MemoryMonitor:
    def __init__(self, threshold: float = 0.8):
        self.threshold = threshold
        self.process = psutil.Process()

    def check_memory(self):
        memory_percent = self.process.memory_percent()
        if memory_percent > self.threshold * 100:
            self.trigger_cleanup()
```
