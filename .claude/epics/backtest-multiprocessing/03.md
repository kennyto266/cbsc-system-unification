### 1. Progress Tracker

Create a `ProgressTracker` class that:
- Tracks task completion percentage
- Calculates ETA based on historical performance
- Manages hierarchical progress (tasks → subtasks)
- Persists progress for crash recovery

```python
class ProgressTracker:
    def __init__(self, total_tasks: int):
        self.total_tasks = total_tasks
        self.completed_tasks = 0
        self.start_time = time.time()
        self.task_times = deque(maxlen=100)

    def update(self, tasks_completed: int):
        # Update progress and recalculate ETA
        self.completed_tasks += tasks_completed
        self.task_times.append(time.time())

    def get_eta(self) -> timedelta:
        # Calculate estimated time remaining
        if not self.task_times:
            return None
        avg_time = sum(self.task_times) / len(self.task_times)
        remaining = self.total_tasks - self.completed_tasks
        return timedelta(seconds=avg_time * remaining)
```

```python
class ResourceMonitor:
    def __init__(self, interval: float = 1.0):
        self.interval = interval
        self.metrics = {}
        self.processes = {}

    def collect_metrics(self):
        # Collect system-wide metrics
        cpu_percent = psutil.cpu_percent(interval=None, percpu=True)
        memory = psutil.virtual_memory()
        disk_io = psutil.disk_io_counters()

        return {
            'cpu': cpu_percent,
            'memory': {
                'total': memory.total,
                'used': memory.used,
                'percent': memory.percent
            },
            'disk_io': disk_io
        }

    def monitor_process(self, pid: int):
        # Monitor specific process metrics
        if pid not in self.processes:
            self.processes[pid] = psutil.Process(pid)

        process = self.processes[pid]
        return {
            'cpu': process.cpu_percent(),
            'memory': process.memory_info(),
            'status': process.status()
        }
```

```python
class ErrorHandler:
    def __init__(self, max_retries: int = 3):
        self.max_retries = max_retries
        self.error_counts = defaultdict(int)
        self.backoff_factor = 2

    def handle_error(self, error: Exception, task: BacktestTask):
        # Determine error type and appropriate action
        if isinstance(error, MemoryError):
            return self._handle_memory_error(task)
        elif isinstance(error, TimeoutError):
            return self._handle_timeout_error(task)
        else:
            return self._handle_generic_error(task)

    def _handle_memory_error(self, task: BacktestTask):
        # Reduce process count and retry
        task.adjust_memory_usage(reduction_factor=0.5)
        return self._retry_task(task, delay=10)

    def _retry_task(self, task: BacktestTask, delay: float = 0):
        # Retry task with exponential backoff
        self.error_counts[task.id] += 1
        if self.error_counts[task.id] < self.max_retries:
            backoff_delay = delay * (self.backoff_factor ** self.error_counts[task.id])
            asyncio.create_task(self._schedule_retry(task, backoff_delay))
            return True
        return False
```
