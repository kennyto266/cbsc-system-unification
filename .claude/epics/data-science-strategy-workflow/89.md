### 1. BacktestAdapter Core Class

Create the main adapter class:

```python
class BacktestAdapter:
    """Adapter for CBSC backtesting API with notebook-friendly interface"""

    def __init__(self, api_base_url: str, auth_token: str):
        self.api_base_url = api_base_url
        self.auth_token = auth_token
        self.client = httpx.AsyncClient()

    async def run_backtest(
        self,
        strategy_code: str,
        symbols: List[str],
        start_date: datetime,
        end_date: datetime,
        parameters: Dict[str, Any],
        on_progress: Optional[Callable[[float], None]] = None
    ) -> BacktestResult:
        """Run backtest and return results"""

    async def get_backtest_status(self, job_id: str) -> BacktestStatus:
        """Check status of running backtest"""

    async def cancel_backtest(self, job_id: str) -> bool:
        """Cancel running backtest"""
```
